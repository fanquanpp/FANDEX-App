# 数据定义

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## CREATE

**CREATE DATABASE：创建数据库**
`CREATE DATABASE [IF NOT EXISTS] <db_name> [CHARACTER SET <charset>] [COLLATE <collation>]`
```sql
-- 创建数据库
CREATE DATABASE mydb;

-- 带字符集和校对规则
CREATE DATABASE mydb
CHARACTER SET utf8mb4
COLLATE utf8mb4_unicode_ci;

-- IF NOT EXISTS 避免报错
CREATE DATABASE IF NOT EXISTS mydb;
```

**CREATE TABLE：创建表**
`CREATE TABLE [IF NOT EXISTS] <table_name> (<列定义> [, ...] [, <约束>])`
```sql
-- 创建表
CREATE TABLE employees (
  id          BIGINT PRIMARY KEY AUTO_INCREMENT,
  name        VARCHAR(50) NOT NULL,
  email       VARCHAR(100) UNIQUE,
  dept_id     BIGINT,
  salary      DECIMAL(10, 2),
  hire_date   DATE,
  created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (dept_id) REFERENCES departments(id)
);

-- IF NOT EXISTS 避免报错
CREATE TABLE IF NOT EXISTS temp_data (
  id INT PRIMARY KEY,
  value TEXT
);
```

**CREATE TABLE AS：从查询结果创建表**
`CREATE TABLE <new_table> AS SELECT ...`
```sql
-- 从查询结果创建表（复制结构和数据）
CREATE TABLE high_salary_employees AS
SELECT * FROM employees WHERE salary > 100000;

-- 只复制结构（不复制数据）
CREATE TABLE employees_backup AS
SELECT * FROM employees WHERE 1=0;
```

**CREATE INDEX：创建索引**
`CREATE [UNIQUE] INDEX <index_name> ON <table>(<列> [, ...])`
```sql
-- 创建普通索引
CREATE INDEX idx_name ON employees(name);

-- 创建唯一索引
CREATE UNIQUE INDEX idx_email ON employees(email);

-- 创建复合索引
CREATE INDEX idx_dept_salary ON employees(dept_id, salary);
```

**CREATE VIEW：创建视图**
`CREATE [OR REPLACE] VIEW <view_name> AS SELECT ...`
```sql
-- 创建视图
CREATE VIEW high_salary_view AS
SELECT name, salary FROM employees WHERE salary > 100000;

-- OR REPLACE 替换已存在的视图
CREATE OR REPLACE VIEW high_salary_view AS
SELECT name, salary, department FROM employees WHERE salary > 80000;
```

## ALTER

**ALTER TABLE：修改表结构**
`ALTER TABLE <table_name> <action>`
```sql
-- 添加列
ALTER TABLE employees ADD COLUMN age INT;

-- 添加多列
ALTER TABLE employees
  ADD COLUMN phone VARCHAR(20),
  ADD COLUMN address TEXT;

-- 删除列
ALTER TABLE employees DROP COLUMN age;

-- 修改列类型
ALTER TABLE employees MODIFY COLUMN name VARCHAR(100);
ALTER TABLE employees ALTER COLUMN name TYPE VARCHAR(100);  -- PostgreSQL

-- 重命名列
ALTER TABLE employees RENAME COLUMN name TO full_name;

-- 重命名表
ALTER TABLE employees RENAME TO staff;
```

**ALTER TABLE 添加约束**
`ALTER TABLE <table> ADD [CONSTRAINT <name>] <约束类型>(<列>)`
```sql
-- 添加主键
ALTER TABLE employees ADD PRIMARY KEY (id);

-- 添加唯一约束
ALTER TABLE employees ADD CONSTRAINT uk_email UNIQUE (email);

-- 添加外键
ALTER TABLE employees
ADD CONSTRAINT fk_dept
FOREIGN KEY (dept_id) REFERENCES departments(id);

-- 添加默认值
ALTER TABLE employees ALTER COLUMN status SET DEFAULT 'active';

-- 删除约束
ALTER TABLE employees DROP CONSTRAINT uk_email;
ALTER TABLE employees DROP INDEX uk_email;  -- MySQL
```

**ALTER DATABASE：修改数据库**
`ALTER DATABASE <db_name> CHARACTER SET <charset>`
```sql
-- 修改数据库字符集
ALTER DATABASE mydb CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
```

**ALTER INDEX：修改索引**
`ALTER INDEX <name> RENAME TO <new_name>`
```sql
-- 重命名索引（PostgreSQL）
ALTER INDEX idx_name RENAME TO idx_employee_name;
```

## DROP

**DROP DATABASE：删除数据库**
`DROP DATABASE [IF EXISTS] <db_name>`
```sql
-- 删除数据库
DROP DATABASE mydb;

-- IF EXISTS 避免报错
DROP DATABASE IF EXISTS mydb;
```

**DROP TABLE：删除表**
`DROP TABLE [IF EXISTS] <table_name> [, ...]`
```sql
-- 删除表
DROP TABLE employees;

-- IF EXISTS 避免报错
DROP TABLE IF EXISTS employees, departments;
```

**DROP INDEX：删除索引**
`DROP INDEX <index_name> ON <table> | DROP INDEX <name>`
```sql
-- MySQL
DROP INDEX idx_name ON employees;

-- PostgreSQL / SQL Server
DROP INDEX idx_name;
```

**DROP VIEW：删除视图**
`DROP VIEW [IF EXISTS] <view_name>`
```sql
-- 删除视图
DROP VIEW high_salary_view;

-- IF EXISTS 避免报错
DROP VIEW IF EXISTS high_salary_view;
```

## TRUNCATE

**TRUNCATE TABLE：清空表数据**
`TRUNCATE TABLE <table_name>`
```sql
-- 清空表数据（保留结构）
TRUNCATE TABLE employees;

-- TRUNCATE vs DELETE:
-- TRUNCATE: DDL，不可回滚，重置自增ID，速度快
-- DELETE: DML，可回滚，不重置自增ID，速度慢
```

## COMMENT

**添加注释：为表或列添加说明**
`COMMENT ON TABLE <table> IS '<注释>' | COMMENT ON COLUMN <table>.<column> IS '<注释>'`
```sql
-- 表注释
COMMENT ON TABLE employees IS '员工信息表';

-- 列注释
COMMENT ON COLUMN employees.salary IS '员工月薪（人民币）';

-- MySQL 简写
CREATE TABLE employees (
  id INT COMMENT '主键ID',
  name VARCHAR(50) COMMENT '员工姓名'
) COMMENT='员工信息表';
```
