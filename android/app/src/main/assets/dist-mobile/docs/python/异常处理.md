## 异常处理基础

**try-except 语句：捕获异常**
`try: <语句>`
`except <异常类型> [as <变量>]: <处理>`

```python
# 基本用法
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")

# 捕获异常信息
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")

# 捕获多种异常
try:
    number = int(input("Enter a number: "))
    result = 10 / number
except ValueError:
    print("Invalid input, please enter a number")
except ZeroDivisionError:
    print("Cannot divide by zero")

# 捕获多种异常（元组形式）
try:
    value = int("abc")
except (ValueError, TypeError) as e:
    print(f"Error: {e}")

# 捕获所有异常
try:
    result = 10 / 0
except Exception as e:
    print(f"An error occurred: {e}")
```

---

**try-except-else-finally：完整结构**
`try: <语句>`
`except <异常类型>: <处理>`
`else: <无异常时执行>`
`finally: <始终执行>`

```python
# 完整结构
try:
    file = open("example.txt", "r")
    content = file.read()
except FileNotFoundError:
    print("File not found")
except PermissionError:
    print("Permission denied")
else:
    print(f"File content: {content}")
finally:
    if 'file' in locals():
        file.close()
    print("Cleanup completed")

# 数据库操作示例
def database_operation():
    connection = None
    try:
        connection = connect_to_database()
        result = connection.execute("SELECT * FROM users")
        return result
    except DatabaseError as e:
        print(f"Database error: {e}")
        return None
    finally:
        if connection:
            connection.close()
```

## 抛出异常

**raise 语句：手动抛出异常**
`raise <异常类型>(<消息>)`

```python
# 抛出异常
def divide(a, b):
    if b == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return a / b

try:
    result = divide(10, 0)
except ZeroDivisionError as e:
    print(f"Error: {e}")

# 抛出异常并附带原始异常
def process_data(data):
    try:
        return int(data)
    except ValueError as e:
        raise ValueError(f"Invalid data: {data}") from e

try:
    process_data("abc")
except ValueError as e:
    print(f"Error: {e}")
    print(f"Original error: {e.__cause__}")

# 重新抛出异常
def risky_operation():
    try:
        result = 10 / 0
    except ZeroDivisionError:
        print("Logging error...")
        raise  # 重新抛出当前异常

# 自定义异常条件
def validate_age(age):
    if age < 0:
        raise ValueError("Age cannot be negative")
    if age > 150:
        raise ValueError("Age cannot be greater than 150")
    return age

try:
    validate_age(-5)
except ValueError as e:
    print(f"Validation error: {e}")
```

## 自定义异常

**自定义异常类：继承 Exception**
`class <异常类>(Exception): def __init__(self, <参数>): ...`

```python
# 基本自定义异常
class CustomError(Exception):
    pass

# 带消息的自定义异常
class ValidationError(Exception):
    def __init__(self, message, field=None):
        self.message = message
        self.field = field
        super().__init__(self.message)
    
    def __str__(self):
        if self.field:
            return f"ValidationError in '{self.field}': {self.message}"
        return f"ValidationError: {self.message}"

# 使用自定义异常
def validate_email(email):
    if "@" not in email:
        raise ValidationError("Invalid email format", "email")
    return True

try:
    validate_email("invalid-email")
except ValidationError as e:
    print(e)  # ValidationError in 'email': Invalid email format

# 异常继承层次
class AppError(Exception):
    """应用程序基础异常"""
    pass

class DatabaseError(AppError):
    """数据库异常"""
    pass

class ConnectionError(DatabaseError):
    """连接异常"""
    pass

class QueryError(DatabaseError):
    """查询异常"""
    pass

# 捕获特定异常
try:
    raise QueryError("Invalid SQL query")
except ConnectionError:
    print("Connection error")
except DatabaseError:
    print("Database error")  # 这里会捕获
except AppError:
    print("Application error")
```

## 异常链

**异常链：使用 from 保留原始异常**
`raise <新异常> from <原始异常>`

```python
# 显式异常链
def parse_config(config_str):
    try:
        import json
        return json.loads(config_str)
    except json.JSONDecodeError as e:
        raise ConfigError("Invalid configuration") from e

class ConfigError(Exception):
    pass

try:
    parse_config("invalid json")
except ConfigError as e:
    print(f"Error: {e}")
    print(f"Caused by: {e.__cause__}")

# 隐式异常链（在异常处理中抛出新异常）
try:
    try:
        1 / 0
    except:
        raise ValueError("New error")
except ValueError as e:
    print(f"Error: {e}")
    print(f"Context: {e.__context__}")  # 原始异常

# 抑制异常链
try:
    try:
        1 / 0
    except:
        raise ValueError("New error") from None  # 抑制异常链
except ValueError as e:
    print(f"Error: {e}")
    print(f"Context: {e.__context__}")  # None
```

## 上下文管理器

**with 语句：自动管理资源**
`with <上下文管理器> [as <变量>]: <语句>`

```python
# 文件操作
with open("example.txt", "r") as file:
    content = file.read()
# 文件自动关闭

# 多个上下文管理器
with open("input.txt", "r") as infile, open("output.txt", "w") as outfile:
    content = infile.read()
    outfile.write(content)

# 数据库连接
with database_connection() as conn:
    conn.execute("SELECT * FROM users")

# 锁
from threading import Lock
lock = Lock()
with lock:
    # 临界区代码
    pass
```

---

**自定义上下文管理器：实现 __enter__ 和 __exit__**
`class <类>: def __enter__(self): ...; def __exit__(self, ...): ...`

```python
class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None
    
    def __enter__(self):
        print(f"Opening file: {self.filename}")
        self.file = open(self.filename, self.mode)
        return self.file
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()
        print(f"File closed: {self.filename}")
        if exc_type:
            print(f"Exception occurred: {exc_val}")
        return False  # 不抑制异常

# 使用自定义上下文管理器
with FileManager("example.txt", "w") as f:
    f.write("Hello, World!")

# 计时器上下文管理器
import time

class Timer:
    def __enter__(self):
        self.start = time.time()
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.end = time.time()
        print(f"Execution time: {self.end - self.start:.4f} seconds")

with Timer():
    time.sleep(1)
    # 执行一些操作
```

**contextlib 模块：使用生成器创建上下文管理器**
`from contextlib import contextmanager`
`@contextmanager def <函数>(): ...`

```python
from contextlib import contextmanager

@contextmanager
def open_file(filename, mode):
    """文件上下文管理器"""
    f = open(filename, mode)
    try:
        yield f
    finally:
        f.close()

with open_file("example.txt", "w") as f:
    f.write("Hello, World!")

@contextmanager
def timer():
    """计时器上下文管理器"""
    import time
    start = time.time()
    try:
        yield
    finally:
        end = time.time()
        print(f"Execution time: {end - start:.4f} seconds")

with timer():
    time.sleep(1)

@contextmanager
def change_directory(path):
    """切换目录上下文管理器"""
    import os
    old_dir = os.getcwd()
    os.chdir(path)
    try:
        yield
    finally:
        os.chdir(old_dir)
```

## 异常处理最佳实践

**捕获特定异常**
`except <具体异常类型>: <处理>`

```python
# 不推荐：捕获所有异常
try:
    result = 10 / 0
except:
    print("An error occurred")

# 推荐：捕获特定异常
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")

# 处理多种具体异常
try:
    value = int(input("Enter a number: "))
    result = 10 / value
    items = [1, 2, 3]
    print(items[value])
except ValueError:
    print("Invalid number")
except ZeroDivisionError:
    print("Cannot divide by zero")
except IndexError:
    print("Index out of range")
```

**记录异常日志**
`import logging`
`logging.exception(<消息>)`

```python
import logging
import traceback

logging.basicConfig(level=logging.ERROR)

def risky_function():
    return 10 / 0

try:
    risky_function()
except Exception as e:
    # 记录异常日志（包含堆栈跟踪）
    logging.error(f"Error occurred: {e}")
    logging.error(traceback.format_exc())

# 使用 logging.exception 自动记录堆栈
try:
    risky_function()
except Exception:
    logging.exception("An error occurred in risky_function")
```
