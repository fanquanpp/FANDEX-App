## 类与对象

**类定义：使用 `class` 关键字**
`class <类名>: <类体>`
`def __init__(self, <参数>): <初始化>`

```python
# 基本类定义
class Dog:
    """狗类"""
    
    # 类变量
    species = "Canis familiaris"
    
    # 初始化方法
    def __init__(self, name, age):
        self.name = name  # 实例变量
        self.age = age
    
    # 实例方法
    def bark(self):
        return f"{self.name} says Woof!"
    
    # __str__ 方法
    def __str__(self):
        return f"Dog(name={self.name}, age={self.age})"

# 创建实例
dog = Dog("Buddy", 3)
print(dog.name)      # Buddy
print(dog.bark())    # Buddy says Woof!
print(dog)           # Dog(name=Buddy, age=3)
print(Dog.species)   # Canis familiaris
```

## 属性与方法

**实例变量：每个对象独立拥有**
`self.<属性> = <值>`

```python
class Person:
    def __init__(self, name, age):
        self.name = name  # 公有属性
        self.age = age
        self._email = ""  # 受保护属性（约定）
        self.__id = 0     # 私有属性（名称重整）

person = Person("Alice", 30)
print(person.name)    # Alice
print(person._email)  # ""（可访问但不推荐）
# print(person.__id)  # AttributeError（不能直接访问）
print(person._Person__id)  # 0（通过名称重整访问）
```

**类变量：所有实例共享**
`class <类名>: <类变量> = <值>`

```python
class Counter:
    """计数器类"""
    count = 0  # 类变量
    
    def __init__(self):
        Counter.count += 1  # 通过类名访问类变量
        self.id = Counter.count  # 实例变量
    
    @classmethod
    def get_count(cls):
        return cls.count

c1 = Counter()
c2 = Counter()
c3 = Counter()
print(Counter.count)  # 3
print(c1.id, c2.id, c3.id)  # 1 2 3
```

**实例方法、类方法、静态方法**
`def <方法>(self, ...)` / `@classmethod def <方法>(cls, ...)` / `@staticmethod def <方法>(...)`

```python
class MathUtils:
    pi = 3.14159265359
    
    # 实例方法
    def __init__(self, value):
        self.value = value
    
    def square(self):
        return self.value ** 2
    
    # 类方法
    @classmethod
    def circle_area(cls, radius):
        return cls.pi * radius ** 2
    
    # 静态方法
    @staticmethod
    def add(a, b):
        return a + b

# 实例方法调用
math_obj = MathUtils(5)
print(math_obj.square())  # 25

# 类方法调用
print(MathUtils.circle_area(2))  # 12.56637061436

# 静态方法调用
print(MathUtils.add(3, 4))  # 7
```

**属性装饰器：getter/setter 控制**
`@property def <属性>(self): ...`
`@<属性>.setter def <属性>(self, value): ...`

```python
class Temperature:
    def __init__(self, celsius):
        self._celsius = celsius
    
    @property
    def celsius(self):
        """获取摄氏温度"""
        return self._celsius
    
    @celsius.setter
    def celsius(self, value):
        """设置摄氏温度"""
        if value < -273.15:
            raise ValueError("Temperature below absolute zero")
        self._celsius = value
    
    @property
    def fahrenheit(self):
        """获取华氏温度（只读）"""
        return self._celsius * 9/5 + 32

temp = Temperature(25)
print(temp.celsius)     # 25
print(temp.fahrenheit)  # 77.0

temp.celsius = 30
print(temp.fahrenheit)  # 86.0

# temp.fahrenheit = 100  # AttributeError（只读属性）
```

## 继承

**单继承：子类继承父类**
`class <子类>(<父类>): <类体>`

```python
# 父类
class Animal:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def speak(self):
        return f"{self.name} makes a sound"
    
    def __str__(self):
        return f"Animal(name={self.name}, age={self.age})"

# 子类
class Dog(Animal):
    def __init__(self, name, age, breed):
        super().__init__(name, age)  # 调用父类初始化方法
        self.breed = breed
    
    def speak(self):  # 方法重写
        return f"{self.name} says Woof!"
    
    def fetch(self):  # 子类特有方法
        return f"{self.name} fetches the ball"

dog = Dog("Buddy", 3, "Golden Retriever")
print(dog.speak())   # Buddy says Woof!
print(dog.fetch())   # Buddy fetches the ball
print(dog.name)      # Buddy（继承自父类）
```

**多继承：继承多个父类**
`class <子类>(<父类1>, <父类2>): <类体>`

```python
class Flyable:
    def fly(self):
        return f"{self.name} is flying"

class Swimmable:
    def swim(self):
        return f"{self.name} is swimming"

class Duck(Flyable, Swimmable):
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        return f"{self.name} says Quack!"

duck = Duck("Donald")
print(duck.fly())   # Donald is flying
print(duck.swim())  # Donald is swimming
print(duck.speak()) # Donald says Quack!

# 查看方法解析顺序（MRO）
print(Duck.__mro__)
# (<class 'Duck'>, <class 'Flyable'>, <class 'Swimmable'>, <class 'object'>)
```

**方法重写与 super()**
`def <方法>(self, ...): super().<方法>(...)`

```python
class Shape:
    def __init__(self, color):
        self.color = color
    
    def area(self):
        return 0
    
    def describe(self):
        return f"A {self.color} shape with area {self.area()}"

class Circle(Shape):
    def __init__(self, color, radius):
        super().__init__(color)
        self.radius = radius
    
    def area(self):  # 重写父类方法
        return 3.14159 * self.radius ** 2

class Square(Shape):
    def __init__(self, color, side):
        super().__init__(color)
        self.side = side
    
    def area(self):
        return self.side ** 2

circle = Circle("red", 5)
square = Square("blue", 4)
print(circle.describe())  # A red shape with area 78.53975
print(square.describe())  # A blue shape with area 16
```

## 特殊方法（魔术方法）

**常用魔术方法：自定义对象行为**
`def __<方法>__(self, ...): ...`

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):  # 字符串表示
        return f"Vector({self.x}, {self.y})"
    
    def __repr__(self):  # 开发者表示
        return f"Vector(x={self.x}, y={self.y})"
    
    def __add__(self, other):  # 加法运算
        return Vector(self.x + other.x, self.y + other.y)
    
    def __sub__(self, other):  # 减法运算
        return Vector(self.x - other.x, self.y - other.y)
    
    def __mul__(self, scalar):  # 乘法运算
        return Vector(self.x * scalar, self.y * scalar)
    
    def __eq__(self, other):  # 等于比较
        return self.x == other.x and self.y == other.y
    
    def __lt__(self, other):  # 小于比较
        return self.magnitude() < other.magnitude()
    
    def __len__(self):  # 长度
        return 2
    
    def __getitem__(self, index):  # 索引访问
        return (self.x, self.y)[index]
    
    def __iter__(self):  # 迭代
        yield self.x
        yield self.y
    
    def __contains__(self, item):  # in 运算
        return item in (self.x, self.y)
    
    def magnitude(self):
        return (self.x ** 2 + self.y ** 2) ** 0.5

v1 = Vector(3, 4)
v2 = Vector(1, 2)

print(v1)           # Vector(3, 4)
print(v1 + v2)     # Vector(4, 6)
print(v1 - v2)     # Vector(2, 2)
print(v1 * 2)      # Vector(6, 8)
print(v1 == Vector(3, 4))  # True
print(len(v1))     # 2
print(v1[0])       # 3
print(list(v1))    # [3, 4]
print(3 in v1)     # True
```

## 抽象基类

**ABC：定义接口规范**
`from abc import ABC, abstractmethod`
`class <类>(ABC): @abstractmethod def <方法>(self): ...`

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    """抽象基类"""
    
    @abstractmethod
    def area(self):
        """计算面积"""
        pass
    
    @abstractmethod
    def perimeter(self):
        """计算周长"""
        pass
    
    def describe(self):
        return f"Area: {self.area()}, Perimeter: {self.perimeter()}"

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        return 3.14159 * self.radius ** 2
    
    def perimeter(self):
        return 2 * 3.14159 * self.radius

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

# shape = Shape()  # TypeError（不能实例化抽象类）
circle = Circle(5)
rectangle = Rectangle(4, 6)
print(circle.describe())    # Area: 78.53975, Perimeter: 31.4159
print(rectangle.describe())  # Area: 24, Perimeter: 20
```

## 多态

**多态：不同对象响应同一方法**
`def <函数>(<对象>: <接口>): <对象>.<方法>()`

```python
class Animal:
    def speak(self):
        pass

class Dog(Animal):
    def speak(self):
        return "Woof!"

class Cat(Animal):
    def speak(self):
        return "Meow!"

class Cow(Animal):
    def speak(self):
        return "Moo!"

def animal_speak(animal):
    """多态函数"""
    print(animal.speak())

animals = [Dog(), Cat(), Cow()]
for animal in animals:
    animal_speak(animal)
# Woof!
# Meow!
# Moo!
```

## 组合

**组合：将对象作为属性**
`class <类>: def __init__(self): self.<组件> = <其他类>()`

```python
class Engine:
    def start(self):
        return "Engine started"
    
    def stop(self):
        return "Engine stopped"

class Wheels:
    def rotate(self):
        return "Wheels rotating"

class Car:
    def __init__(self, brand):
        self.brand = brand
        self.engine = Engine()  # 组合
        self.wheels = Wheels()  # 组合
    
    def start(self):
        return f"{self.brand}: {self.engine.start()}, {self.wheels.rotate()}"

car = Car("Toyota")
print(car.start())  # Toyota: Engine started, Wheels rotating
```

## 数据类

**dataclass：自动生成常用方法**
`from dataclasses import dataclass`
`@dataclass class <类>: <字段>: <类型>`

```python
from dataclasses import dataclass, field
from typing import List

@dataclass
class Person:
    name: str
    age: int
    city: str = "Unknown"
    hobbies: List[str] = field(default_factory=list)
    
    def greet(self):
        return f"Hello, I'm {self.name}"

# 自动生成 __init__, __repr__, __eq__
person1 = Person("Alice", 30, "New York", ["reading", "swimming"])
person2 = Person("Alice", 30, "New York", ["reading", "swimming"])
print(person1)              # Person(name='Alice', age=30, city='New York', hobbies=['reading', 'swimming'])
print(person1 == person2)  # True
print(person1.greet())      # Hello, I'm Alice

# 不可变数据类
@dataclass(frozen=True)
class Point:
    x: float
    y: float

p = Point(3.0, 4.0)
print(p)  # Point(x=3.0, y=4.0)
# p.x = 5.0  # FrozenInstanceError（不可修改）
```

## 元类

**元类：创建类的类**
`class <元类>(type): def __new__(mcs, name, bases, namespace): ...`
`class <类>(metaclass=<元类>): ...`

```python
class SingletonMeta(type):
    """单例元类"""
    _instances = {}
    
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=SingletonMeta):
    def __init__(self):
        self.connection = "connected"

db1 = Database()
db2 = Database()
print(db1 is db2)  # True（同一实例）
```

## 魔术方法汇总

**运算符重载方法**
`def __<运算符>__(self, other): ...`

```python
class Number:
    def __init__(self, value):
        self.value = value
    
    # 算术运算
    def __add__(self, other): return Number(self.value + other.value)
    def __sub__(self, other): return Number(self.value - other.value)
    def __mul__(self, other): return Number(self.value * other.value)
    def __truediv__(self, other): return Number(self.value / other.value)
    
    # 比较运算
    def __eq__(self, other): return self.value == other.value
    def __lt__(self, other): return self.value < other.value
    def __gt__(self, other): return self.value > other.value
    
    # 一元运算
    def __neg__(self): return Number(-self.value)
    def __abs__(self): return Number(abs(self.value))
    
    # 类型转换
    def __int__(self): return int(self.value)
    def __float__(self): return float(self.value)
    
    def __str__(self): return str(self.value)
    def __repr__(self): return f"Number({self.value})"
```
