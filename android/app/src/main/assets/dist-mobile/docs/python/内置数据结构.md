# 内置数据结构

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 列表（List）

**列表创建：有序可变序列**
`<列表> = [<元素1>, <元素2>, ...]`
`list(<可迭代对象>)`

```python
# 直接创建
fruits = ["apple", "banana", "cherry"]
numbers = [1, 2, 3, 4, 5]
mixed = [1, "hello", 3.14, True, [1, 2, 3]]

# 使用 list() 函数
empty_list = list()
from_iter = list("hello")  # ['h', 'e', 'l', 'l', 'o']
from_range = list(range(5))  # [0, 1, 2, 3, 4]

# 列表生成式
squares = [x ** 2 for x in range(5)]  # [0, 1, 4, 9, 16]
```

---

**列表索引与切片：访问元素**
`<列表>[<索引>]`
`<列表>[<start>:<stop>:<step>]`

```python
fruits = ["apple", "banana", "cherry", "date", "elderberry"]

# 索引访问
print(fruits[0])    # 'apple'
print(fruits[-1])   # 'elderberry'

# 切片
print(fruits[1:3])    # ['banana', 'cherry']
print(fruits[:2])     # ['apple', 'banana']
print(fruits[2:])     # ['cherry', 'date', 'elderberry']
print(fruits[::2])    # ['apple', 'cherry', 'elderberry']
print(fruits[::-1])   # 反转列表

# 修改元素
fruits[0] = "apricot"
fruits[1:3] = ["blueberry", "cranberry"]
```

---

**列表方法：增删改查**
`<列表>.<方法>(<参数>)`

```python
fruits = ["apple", "banana"]

# 添加元素
fruits.append("cherry")        # 末尾添加
fruits.insert(1, "blueberry")  # 指定位置插入
fruits.extend(["date", "elderberry"])  # 扩展列表
print(fruits)

# 删除元素
fruits.remove("banana")   # 按值删除
popped = fruits.pop()     # 弹出末尾元素
popped = fruits.pop(0)    # 弹出指定位置
del fruits[0]             # 按索引删除

# 查找
index = fruits.index("cherry")  # 查找索引
count = fruits.count("apple")    # 计数

# 排序
fruits.sort()              # 原地排序
fruits.sort(reverse=True)  # 降序
fruits.sort(key=len)       # 按长度排序
sorted_fruits = sorted(fruits)  # 返回新列表

# 反转
fruits.reverse()  # 原地反转
reversed_fruits = list(reversed(fruits))  # 返回新列表

# 复制
copy_list = fruits.copy()
copy_list = fruits[:]
```

## 元组（Tuple）

**元组创建：有序不可变序列**
`<元组> = (<元素1>, <元素2>, ...)`
`tuple(<可迭代对象>)`

```python
# 直接创建
point = (3, 4)
person = ("Alice", 30, "New York")
single_tuple = (42,)  # 单元素元组需加逗号
empty_tuple = ()

# 使用 tuple() 函数
from_list = tuple([1, 2, 3])
from_string = tuple("hello")  # ('h', 'e', 'l', 'l', 'o')

# 元组解包
x, y = point
name, age, city = person

# 扩展解包
first, *rest = range(5)  # first=0, rest=[1, 2, 3, 4]
*init, last = range(5)   # init=[0, 1, 2, 3], last=4
first, *middle, last = range(5)  # first=0, middle=[1, 2, 3], last=4
```

---

**元组方法：仅支持查找**
`<元组>.<方法>(<参数>)`

```python
numbers = (1, 2, 3, 2, 4, 2)

# 查找
print(numbers.count(2))  # 3（计数）
print(numbers.index(3))  # 2（首次出现的索引）

# 元组不可变，但内部可变对象可修改
nested = (1, [2, 3], 4)
nested[1].append(5)
print(nested)  # (1, [2, 3, 5], 4)
```

## 字典（Dict）

**字典创建：键值对映射**
`<字典> = {<键1>: <值1>, <键2>: <值2>}`
`dict(<键值对序列>)`

```python
# 直接创建
person = {
    "name": "Alice",
    "age": 30,
    "city": "New York"
}

# 使用 dict() 函数
person2 = dict(name="Bob", age=25, city="London")
person3 = dict([("name", "Charlie"), ("age", 35)])

# 字典推导式
squares = {x: x ** 2 for x in range(5)}  # {0: 0, 1: 1, 2: 4, 3: 9, 4: 16}

# 使用 fromkeys() 创建字典
keys = ["a", "b", "c"]
default_dict = dict.fromkeys(keys, 0)  # {'a': 0, 'b': 0, 'c': 0}
```

---

**字典访问与修改**
`<字典>[<键>]`
`<字典>.get(<键>, [<默认值>])`

```python
person = {"name": "Alice", "age": 30}

# 访问
print(person["name"])           # Alice
print(person.get("age"))         # 30
print(person.get("city", "Unknown"))  # Unknown（键不存在时返回默认值）

# 修改和添加
person["age"] = 31           # 修改
person["city"] = "Boston"   # 添加
person.update({"age": 32, "country": "USA"})  # 批量更新

# 删除
del person["city"]           # 删除指定键
age = person.pop("age")      # 弹出并返回值
person.clear()               # 清空字典
```

---

**字典遍历**
`<字典>.keys()` / `<字典>.values()` / `<字典>.items()`

```python
person = {"name": "Alice", "age": 30, "city": "New York"}

# 遍历键
for key in person.keys():
    print(key)

# 遍历值
for value in person.values():
    print(value)

# 遍历键值对
for key, value in person.items():
    print(f"{key}: {value}")

# 字典推导式
prices = {"apple": 1.5, "banana": 0.5, "cherry": 3.0}
expensive = {k: v for k, v in prices.items() if v > 1.0}
print(expensive)  # {'apple': 1.5, 'cherry': 3.0}
```

## 集合（Set）

**集合创建：无序不重复元素集**
`<集合> = {<元素1>, <元素2>, ...}`
`set(<可迭代对象>)`

```python
# 直接创建
fruits = {"apple", "banana", "cherry"}
numbers = {1, 2, 3, 4, 5}

# 使用 set() 函数
from_list = set([1, 2, 2, 3, 3, 4])  # {1, 2, 3, 4}（去重）
from_string = set("hello")  # {'h', 'e', 'l', 'o'}

# 空集合必须用 set()
empty_set = set()  # 不能用 {}（这是空字典）

# 集合推导式
squares = {x ** 2 for x in range(5)}  # {0, 1, 4, 9, 16}
```

---

**集合运算：并集、交集、差集、对称差集**
`<集合> | <集合>` / `<集合> & <集合>` / `<集合> - <集合>` / `<集合> ^ <集合>`

```python
a = {1, 2, 3, 4, 5}
b = {4, 5, 6, 7, 8}

# 并集
union = a | b           # {1, 2, 3, 4, 5, 6, 7, 8}
union = a.union(b)      # 等价写法

# 交集
intersection = a & b              # {4, 5}
intersection = a.intersection(b)  # 等价写法

# 差集
difference = a - b             # {1, 2, 3}
difference = a.difference(b)   # 等价写法

# 对称差集（仅在其中一个集合中出现的元素）
symmetric_diff = a ^ b                    # {1, 2, 3, 6, 7, 8}
symmetric_diff = a.symmetric_difference(b)  # 等价写法

# 子集和超集
c = {1, 2, 3}
print(c.issubset(a))     # True（c 是 a 的子集）
print(a.issuperset(c))    # True（a 是 c 的超集）
print(c <= a)             # True
print(a >= c)             # True

# 不相交
print(a.isdisjoint({10, 11}))  # True（无交集）
```

---

**集合方法：增删**
`<集合>.<方法>(<参数>)`

```python
fruits = {"apple", "banana"}

# 添加
fruits.add("cherry")           # 添加单个元素
fruits.update(["date", "elderberry"])  # 添加多个元素

# 删除
fruits.remove("banana")  # 不存在时抛出 KeyError
fruits.discard("banana")  # 不存在时不报错
popped = fruits.pop()    # 随机弹出
fruits.clear()           # 清空集合
```

## 数据结构转换

**容器类型互转**
`list(<可迭代对象>)` / `tuple(<可迭代对象>)` / `set(<可迭代对象>)` / `dict(<键值对序列>)`

```python
# 列表转元组
my_list = [1, 2, 3]
my_tuple = tuple(my_list)  # (1, 2, 3)

# 元组转列表
my_tuple = (1, 2, 3)
my_list = list(my_tuple)  # [1, 2, 3]

# 列表转集合（去重）
my_list = [1, 2, 2, 3, 3, 4]
my_set = set(my_list)  # {1, 2, 3, 4}

# 集合转列表
my_set = {1, 2, 3}
my_list = list(my_set)  # [1, 2, 3]（顺序不保证）

# 键值对列表转字典
pairs = [("a", 1), ("b", 2), ("c", 3)]
my_dict = dict(pairs)  # {'a': 1, 'b': 2, 'c': 3}

# 字典转键值对列表
my_dict = {"a": 1, "b": 2}
pairs = list(my_dict.items())  # [('a', 1), ('b', 2)]
```

## collections 模块

**namedtuple：具名元组**
`from collections import namedtuple`
`<类型> = namedtuple("<类型名>", [<字段名>])`

```python
from collections import namedtuple

# 创建具名元组类型
Point = namedtuple("Point", ["x", "y"])
Person = namedtuple("Person", ["name", "age", "city"])

# 创建实例
p = Point(3, 4)
person = Person(name="Alice", age=30, city="New York")

# 访问字段
print(p.x, p.y)            # 3 4
print(person.name)          # Alice
print(person[0])            # Alice（也支持索引访问）

# 转换为字典
print(person._asdict())  # {'name': 'Alice', 'age': 30, 'city': 'New York'}

# 替换字段（返回新实例）
new_person = person._replace(age=31)
```

---

**deque：双端队列**
`from collections import deque`
`deque(<可迭代对象>, [maxlen=<长度>])`

```python
from collections import deque

# 创建双端队列
dq = deque([1, 2, 3])
dq = deque(maxlen=3)  # 限制长度

# 两端操作
dq.append(4)         # 右端添加
dq.appendleft(0)     # 左端添加
dq.pop()             # 右端弹出
dq.popleft()         # 左端弹出

# 扩展
dq.extend([5, 6])         # 右端扩展
dq.extendleft([1, 2])     # 左端扩展

# 旋转
dq = deque([1, 2, 3, 4, 5])
dq.rotate(2)   # 右旋: deque([4, 5, 1, 2, 3])
dq.rotate(-1)  # 左旋: deque([5, 1, 2, 3, 4])
```

**Counter：计数器**
`from collections import Counter`
`Counter(<可迭代对象>)`

```python
from collections import Counter

# 创建计数器
text = "hello world"
counter = Counter(text)
print(counter)  # Counter({'l': 3, 'o': 2, 'h': 1, 'e': 1, ' ': 1, 'w': 1, 'r': 1, 'd': 1})

# 计数列表
numbers = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
counter = Counter(numbers)
print(counter[3])  # 3

# 最常见元素
print(counter.most_common(2))  # [(4, 4), (3, 3)]

# 数学运算
c1 = Counter(a=3, b=1)
c2 = Counter(a=1, b=2)
print(c1 + c2)  # Counter({'a': 4, 'b': 3})
print(c1 - c2)  # Counter({'a': 2})
print(c1 & c2)  # Counter({'a': 1, 'b': 1})（交集，取最小）
print(c1 | c2)  # Counter({'a': 3, 'b': 2})（并集，取最大）
```

---

**defaultdict：带默认值的字典**
`from collections import defaultdict`
`defaultdict(<默认工厂>, [<初始值>])`

```python
from collections import defaultdict

# 使用 list 作为默认工厂
words_by_length = defaultdict(list)
words = ["apple", "banana", "cherry", "date"]
for word in words:
    words_by_length[len(word)].append(word)
print(words_by_length)  # {5: ['apple'], 6: ['banana', 'cherry'], 4: ['date']}

# 使用 int 作为默认工厂（计数）
counter = defaultdict(int)
for char in "hello":
    counter[char] += 1
print(counter)  # defaultdict(<class 'int'>, {'h': 1, 'e': 1, 'l': 2, 'o': 1})

# 使用 set 作为默认工厂
groups = defaultdict(set)
groups["fruits"].add("apple")
groups["fruits"].add("banana")
groups["vegetables"].add("carrot")
```

**OrderedDict：有序字典（Python 3.7+ dict 已有序）**
`from collections import OrderedDict`
`OrderedDict(<键值对序列>)`

```python
from collections import OrderedDict

# 创建有序字典
od = OrderedDict()
od["a"] = 1
od["b"] = 2
od["c"] = 3

# 按插入顺序遍历
for key, value in od.items():
    print(f"{key}: {value}")

# 移动到末尾
od.move_to_end("a")

# 弹出首尾
first_item = od.popitem(last=False)  # 弹出首项
last_item = od.popitem(last=True)     # 弹出尾项
```

## 数据结构选择

**根据需求选择数据结构**
`<数据结构> = <合适类型>`

```python
# 需要有序、可变序列 → 列表
shopping_list = ["milk", "bread", "eggs"]
shopping_list.append("butter")

# 需要有序、不可变序列 → 元组
coordinates = (40.7128, -74.0060)

# 需要键值对映射 → 字典
user = {"name": "Alice", "age": 30}

# 需要去重 → 集合
unique_numbers = {1, 2, 3, 4, 5}

# 需要快速成员检查 → 集合
valid_emails = {"alice@example.com", "bob@example.com"}
if "alice@example.com" in valid_emails:
    print("Valid email")

# 需要保持插入顺序且去重 → dict（Python 3.7+）
unique_ordered = list(dict.fromkeys([3, 1, 2, 3, 1, 2]))
print(unique_ordered)  # [3, 1, 2]
```
