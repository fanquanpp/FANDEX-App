# 指针深度解析

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 指针定义与初始化

**指针定义**
`<type> *<pointer_name>;`
```c
int *p;        // 整型指针
float *fp;     // 浮点型指针
char *cp;      // 字符型指针
void *vp;      // 通用指针（可以指向任何类型）
```

---

**指针初始化**
`<type> *<ptr> = &<var>;` 或 `<type> *<ptr> = NULL;`
```c
int a = 10;
int *p = &a;       // p 存储 a 的地址
int *p2 = NULL;    // 空指针
int *p3 = p;       // p3 指向与 p 相同的地址
```

---

**指针解引用**
`*<ptr>`
```c
int a = 10;
int *p = &a;
printf("%d\n", *p);    // 输出 10（解引用）
*p = 20;               // 通过指针修改 a 的值
printf("%d\n", a);     // 输出 20
```

---

## 指针运算

**指针加法**
`<ptr> + <n>`
```c
int arr[] = {10, 20, 30, 40, 50};
int *p = arr;          // 指向 arr[0]
p = p + 1;             // 移动到下一个元素
printf("%d\n", *p);    // 输出 20
p = p + 2;             // 移动 2 个元素
printf("%d\n", *p);    // 输出 40
```

---

**指针减法**
`<ptr> - <n>`
```c
int arr[] = {10, 20, 30, 40, 50};
int *p = &arr[4];      // 指向 arr[4]
p = p - 1;             // 移动到前一个元素
printf("%d\n", *p);    // 输出 40
```

---

**指针比较**
`<ptr1> <==|!=|>|<|>=|<=> <ptr2>`
```c
int arr[] = {10, 20, 30};
int *p1 = &arr[0];
int *p2 = &arr[2];
if (p1 < p2) {
    printf("p1 在 p2 的前面\n");
}
if (p1 == &arr[0]) {
    printf("p1 指向第一个元素\n");
}
```

---

**指针差值**
`<ptr1> - <ptr2>`
```c
int arr[] = {10, 20, 30, 40, 50};
int *p1 = &arr[0];
int *p2 = &arr[3];
int diff = p2 - p1;
printf("元素个数: %d\n", diff);    // 输出 3
```

## 指针与数组

**数组名与指针**
`<array_name>` 等同于 `&<array_name>[0]`
```c
int arr[5] = {1, 2, 3, 4, 5};
int *p = arr;              // 等同于 int *p = &arr[0];
printf("%d\n", arr[2]);     // 使用数组下标
printf("%d\n", *(p + 2));   // 使用指针
```

---

**指针遍历数组**
`for (...) { *<ptr>; <ptr>++; }`
```c
int arr[] = {1, 2, 3, 4, 5};
int *p = arr;
int size = sizeof(arr) / sizeof(arr[0]);
for (int i = 0; i < size; i++) {
    printf("%d ", *p);
    p++;
}
```

## 指针数组与数组指针

**指针数组：存储指针的数组**
`<type> *<array_name>[<size>];`
```c
int *ptr_array[3];
int a = 10, b = 20, c = 30;
ptr_array[0] = &a;
ptr_array[1] = &b;
ptr_array[2] = &c;
// 字符串数组
char *str_array[] = {"Hello", "World", "C"};
```

---

**数组指针：指向数组的指针**
`<type> (*<pointer_name>)[<size>];`
```c
int arr[3] = {1, 2, 3};
int (*p)[3] = &arr;          // 指向整个数组
printf("%d\n", *(*p));       // 输出 1
printf("%d\n", *(*p + 1));   // 输出 2
printf("%d\n", (*p)[2]);     // 输出 3
```

## 多级指针

**二级指针**
`<type> **<pointer_name>;`
```c
int a = 10;
int *p = &a;          // 一级指针
int **pp = &p;        // 二级指针
printf("%d\n", a);     // 输出 10
printf("%d\n", *p);    // 输出 10
printf("%d\n", **pp);  // 输出 10
**pp = 20;            // 通过二级指针修改 a
printf("%d\n", a);     // 输出 20
```

---

**动态二维数组**
`<type> **<matrix> = (<type> **)malloc(<rows> * sizeof(<type> *));`
```c
int rows = 3, cols = 4;
int **matrix = (int **)malloc(rows * sizeof(int *));
for (int i = 0; i < rows; i++) {
    matrix[i] = (int *)malloc(cols * sizeof(int));
}
// 使用
for (int i = 0; i < rows; i++) {
    for (int j = 0; j < cols; j++) {
        matrix[i][j] = i * cols + j;
    }
}
// 释放
for (int i = 0; i < rows; i++) {
    free(matrix[i]);
}
free(matrix);
```

## 函数指针

**函数指针定义**
`return_type (*<pointer_name>)(<parameter_list>);`
```c
int add(int a, int b) { return a + b; }
// 定义函数指针
int (*func_ptr)(int, int);
// 赋值
func_ptr = add;
// 或直接初始化
int (*func_ptr)(int, int) = add;
```

---

**通过函数指针调用**
`<result> = <func_ptr>(<args>);` 或 `<result> = (*<func_ptr>)(<args>);`
```c
int result = func_ptr(10, 20);          // 调用
int result2 = (*func_ptr)(10, 20);      // 更明确的写法
```

---

**回调函数**
`typedef return_type (*<TypeName>)(<params>);`
```c
typedef int (*CompareFunc)(int, int);
void sort(int arr[], int size, CompareFunc compare) {
    for (int i = 0; i < size - 1; i++) {
        for (int j = 0; j < size - i - 1; j++) {
            if (compare(arr[j], arr[j + 1]) > 0) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
int ascending(int a, int b) { return a - b; }
int descending(int a, int b) { return b - a; }
```

## void 指针

**void 指针使用**
`void *<ptr>;`
```c
void *generic_ptr;
int a = 10;
char c = 'A';
// 指向整型
generic_ptr = &a;
printf("%d\n", *(int *)generic_ptr);     // 先转换为 int* 再解引用
// 指向字符
generic_ptr = &c;
printf("%c\n", *(char *)generic_ptr);     // 先转换为 char* 再解引用
```

## 动态内存管理

**malloc：分配未初始化内存**
`void *malloc(size_t <size>);`
```c
int *p = (int *)malloc(5 * sizeof(int));
```

---

**calloc：分配并初始化为零**
`void *calloc(size_t <num>, size_t <size>);`
```c
int *p = (int *)calloc(5, sizeof(int));    // 所有元素初始化为 0
```

---

**realloc：重新分配内存大小**
`void *realloc(void *<ptr>, size_t <new_size>);`
```c
int *new_p = (int *)realloc(p, 10 * sizeof(int));
```

---

**free：释放内存**
`void free(void *<ptr>);`
```c
free(p);
p = NULL;    // 避免野指针
```

## 链表实现

**链表节点定义与操作**
`typedef struct <Node> { <type> data; struct <Node> *next; } <Node>;`
```c
typedef struct Node {
    int data;
    struct Node *next;
} Node;
// 创建新节点
Node *create_node(int data) {
    Node *new_node = (Node *)malloc(sizeof(Node));
    if (new_node == NULL) return NULL;
    new_node->data = data;
    new_node->next = NULL;
    return new_node;
}
// 添加节点到链表末尾
void append(Node **head, int data) {
    Node *new_node = create_node(data);
    if (*head == NULL) { *head = new_node; return; }
    Node *temp = *head;
    while (temp->next != NULL) { temp = temp->next; }
    temp->next = new_node;
}
// 释放链表
void free_list(Node *head) {
    Node *temp;
    while (head != NULL) {
        temp = head;
        head = head->next;
        free(temp);
    }
}
```
