# 动态内存管理

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 内存分配

**malloc：分配未初始化内存**
`void *malloc(size_t <size>);`
```c
#include <stdlib.h>
// 分配 5 个 int 的内存
int *arr = (int *)malloc(5 * sizeof(int));
if (arr == NULL) {
    fprintf(stderr, "Memory allocation failed\n");
    return 1;
}
// 使用
for (int i = 0; i < 5; i++) {
    arr[i] = i + 1;
}
```

---

**calloc：分配并初始化为零**
`void *calloc(size_t <num>, size_t <size>);`
```c
// 分配 5 个 int 的内存，所有元素初始化为 0
int *arr = (int *)calloc(5, sizeof(int));
if (arr == NULL) {
    fprintf(stderr, "Memory allocation failed\n");
    return 1;
}
// 使用
for (int i = 0; i < 5; i++) {
    printf("%d ", arr[i]);    // 输出: 0 0 0 0 0
}
```

---

**realloc：重新分配内存大小**
`void *realloc(void *<ptr>, size_t <new_size>);`
```c
int *arr = (int *)malloc(5 * sizeof(int));
// 扩展到 10 个元素
int *new_arr = (int *)realloc(arr, 10 * sizeof(int));
if (new_arr == NULL) {
    free(arr);    // 原内存仍有效
    return 1;
}
arr = new_arr;
// 使用新内存
for (int i = 5; i < 10; i++) {
    arr[i] = i + 1;
}
```

---

**free：释放内存**
`void free(void *<ptr>);`
```c
int *arr = (int *)malloc(5 * sizeof(int));
// 使用完毕
free(arr);
arr = NULL;    // 避免野指针
```

## 动态数组

**动态数组实现**
`<type> *<arr> = (<type> *)malloc(<size> * sizeof(<type>));`
```c
typedef struct {
    int *data;
    size_t size;
    size_t capacity;
} DynamicArray;
// 初始化
void init_array(DynamicArray *arr, size_t initial_capacity) {
    arr->data = (int *)malloc(initial_capacity * sizeof(int));
    arr->size = 0;
    arr->capacity = initial_capacity;
}
// 添加元素
void push_back(DynamicArray *arr, int value) {
    if (arr->size >= arr->capacity) {
        arr->capacity *= 2;
        arr->data = (int *)realloc(arr->data, arr->capacity * sizeof(int));
    }
    arr->data[arr->size++] = value;
}
// 释放
void free_array(DynamicArray *arr) {
    free(arr->data);
    arr->data = NULL;
    arr->size = 0;
    arr->capacity = 0;
}
```

## 动态字符串

**动态字符串实现**
`char *<str> = (char *)malloc(<size>);`
```c
typedef struct {
    char *data;
    size_t length;
    size_t capacity;
} DynamicString;
// 创建字符串
DynamicString *create_string(const char *initial) {
    DynamicString *str = (DynamicString *)malloc(sizeof(DynamicString));
    str->length = strlen(initial);
    str->capacity = str->length + 1;
    str->data = (char *)malloc(str->capacity);
    strcpy(str->data, initial);
    return str;
}
// 追加字符串
void append_string(DynamicString *str, const char *suffix) {
    size_t new_length = str->length + strlen(suffix);
    if (new_length + 1 > str->capacity) {
        str->capacity = new_length * 2 + 1;
        str->data = (char *)realloc(str->data, str->capacity);
    }
    strcat(str->data, suffix);
    str->length = new_length;
}
// 释放
void free_string(DynamicString *str) {
    free(str->data);
    free(str);
}
```

## 动态二维数组

**使用指针数组实现**
`<type> **<matrix> = (<type> **)malloc(<rows> * sizeof(<type> *));`
```c
int rows = 3, cols = 4;
int **matrix = (int **)malloc(rows * sizeof(int *));
for (int i = 0; i < rows; i++) {
    matrix[i] = (int *)malloc(cols * sizeof(int));
}
// 使用
for (int i = 0; i < rows; i++) {
    for (int j = 0; j < cols; j++) {
        matrix[i][j] = i * cols + j;
    }
}
// 释放
for (int i = 0; i < rows; i++) {
    free(matrix[i]);
}
free(matrix);
```

---

**使用一维数组模拟二维数组**
`<type> *<matrix> = (<type> *)malloc(<rows> * <cols> * sizeof(<type>));`
```c
int rows = 3, cols = 4;
int *matrix = (int *)malloc(rows * cols * sizeof(int));
// 访问 matrix[i][j] 使用 matrix[i * cols + j]
for (int i = 0; i < rows; i++) {
    for (int j = 0; j < cols; j++) {
        matrix[i * cols + j] = i * cols + j;
    }
}
// 释放
free(matrix);
```

## 动态结构体数组

**分配结构体数组**
`<StructType> *<arr> = (<StructType> *)malloc(<count> * sizeof(<StructType>));`
```c
typedef struct {
    int id;
    char name[50];
    float score;
} Student;
// 分配 10 个 Student
Student *students = (Student *)malloc(10 * sizeof(Student));
if (students == NULL) {
    return 1;
}
// 使用
for (int i = 0; i < 10; i++) {
    students[i].id = i + 1;
    sprintf(students[i].name, "Student_%d", i + 1);
    students[i].score = 80.0 + i;
}
// 释放
free(students);
```

## 内存泄漏检测

**自定义内存跟踪**
`void *<tracked_malloc>(size_t <size>, const char *<file>, int <line>);`
```c
// 自定义 malloc 宏
#define TRACK_MALLOC(size) tracked_malloc(size, __FILE__, __LINE__)
void *tracked_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    printf("[ALLOC] %p (%zu bytes) at %s:%d\n", ptr, size, file, line);
    return ptr;
}
// 自定义 free 宏
#define TRACK_FREE(ptr) tracked_free(ptr, __FILE__, __LINE__)
void tracked_free(void *ptr, const char *file, int line) {
    printf("[FREE] %p at %s:%d\n", ptr, file, line);
    free(ptr);
}
// 使用
int *data = (int *)TRACK_MALLOC(10 * sizeof(int));
TRACK_FREE(data);
```

## 内存管理最佳实践

**检查分配结果**
`if (<ptr> == NULL) { /* 处理错误 */ }`
```c
int *ptr = (int *)malloc(size * sizeof(int));
if (ptr == NULL) {
    fprintf(stderr, "Memory allocation failed\n");
    exit(EXIT_FAILURE);
}
```

---

**安全释放**
`free(<ptr>); <ptr> = NULL;`
```c
free(ptr);
ptr = NULL;    // 避免野指针
// 释放后检查
if (ptr != NULL) {
    free(ptr);
    ptr = NULL;
}
```

---

**realloc 安全使用**
`<type> *<new_ptr> = (<type> *)realloc(<ptr>, <new_size>);`
```c
int *new_ptr = (int *)realloc(ptr, new_size * sizeof(int));
if (new_ptr == NULL) {
    // 分配失败，原内存仍有效
    free(ptr);
    return 1;
}
ptr = new_ptr;    // 更新指针
```
