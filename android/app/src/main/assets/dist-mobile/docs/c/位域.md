# 位域

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 位域定义

**基本位域定义**
`struct <Name> { <type> <member> : <width>; };`
```c
struct Flags {
    unsigned int is_active : 1;     // 1 位
    unsigned int is_admin : 1;      // 1 位
    unsigned int level : 3;         // 3 位（0-7）
    unsigned int priority : 4;      // 4 位（0-15）
};
// 使用
struct Flags flags = {0};
flags.is_active = 1;
flags.is_admin = 0;
flags.level = 5;
flags.priority = 10;
```

---

**位域类型选择**
`<unsigned int|signed int|_Bool|...> <member> : <width>;`
```c
struct MixedFlags {
    unsigned int flag1 : 1;     // 无符号 1 位
    signed int flag2 : 2;       // 有符号 2 位（-2 到 1）
    _Bool flag3 : 1;            // 布尔位域（C99+）
    unsigned int value : 8;     // 8 位无符号（0-255）
};
```

## 位域访问

**位域成员访问**
`<var>.<member>` 或 `<ptr>-><member>`
```c
struct Status {
    unsigned int ready : 1;
    unsigned int error : 1;
    unsigned int count : 6;
};
struct Status s = {0};
s.ready = 1;        // 设置 ready 位
s.error = 0;        // 清除 error 位
s.count = 42;       // 设置 count 值
// 通过指针访问
struct Status *ptr = &s;
ptr->ready = 0;
```

---

**位域溢出处理**
`<member> = <value>; // value 超出范围时截断`
```c
struct Small {
    unsigned int val : 3;    // 范围 0-7
};
struct Small s;
s.val = 7;     // 正常
s.val = 8;     // 溢出，存储为 0
s.val = 9;     // 溢出，存储为 1
```

## 位域与联合体配合

**联合体访问位域整体值**
`union <Name> { struct <Bits> <bits>; <type> <value>; };`
```c
union ByteFlags {
    struct {
        unsigned char bit0 : 1;
        unsigned char bit1 : 1;
        unsigned char bit2 : 1;
        unsigned char bit3 : 1;
        unsigned char bit4 : 1;
        unsigned char bit5 : 1;
        unsigned char bit6 : 1;
        unsigned char bit7 : 1;
    } bits;
    unsigned char value;
};
// 使用
union ByteFlags bf;
bf.value = 0;          // 清空所有位
bf.bits.bit0 = 1;       // 设置第 0 位
bf.bits.bit3 = 1;       // 设置第 3 位
printf("Value: 0x%02X\n", bf.value);    // 输出: 0x09
// 通过整体值修改
bf.value = 0xFF;        // 所有位置 1
printf("bit0: %d\n", bf.bits.bit0);     // 输出: 1
```

---

**硬件寄存器映射**
`typedef union { struct { <bitfields> }; <type> <reg>; } <RegType>;`
```c
// 模拟硬件寄存器
typedef union {
    struct {
        unsigned int enable : 1;
        unsigned int mode : 2;
        unsigned int speed : 3;
        unsigned int : 2;          // 保留位
    } bits;
    unsigned int value;
} ControlRegister;
// 使用
ControlRegister reg;
reg.value = 0;
reg.bits.enable = 1;
reg.bits.mode = 2;       // 模式 2
reg.bits.speed = 5;      // 速度 5
printf("Register: 0x%08X\n", reg.value);
```

## 位操作宏

**位操作宏定义**
`#define <NAME>(x, pos) ((x) <op> (1U << (pos)))`
```c
// 检查某一位
#define CHECK_BIT(x, pos)  ((x) & (1U << (pos)))
// 设置某一位为 1
#define SET_BIT(x, pos)    ((x) |= (1U << (pos)))
// 清除某一位为 0
#define CLEAR_BIT(x, pos)  ((x) &= ~(1U << (pos)))
// 切换某一位
#define TOGGLE_BIT(x, pos) ((x) ^= (1U << (pos)))
// 使用
unsigned int value = 0;
SET_BIT(value, 2);        // value = 0x04
SET_BIT(value, 5);        // value = 0x24
if (CHECK_BIT(value, 2)) { /* bit 2 is set */ }
TOGGLE_BIT(value, 2);     // value = 0x20
CLEAR_BIT(value, 5);      // value = 0x00
```

---

**多字节位操作宏**
`#define <NAME>(x, pos, width) (((x) >> (pos)) & ((1U << (width)) - 1))`
```c
// 获取从 pos 开始的 width 位
#define GET_BITS(x, pos, width) \
    (((x) >> (pos)) & ((1U << (width)) - 1))
// 设置从 pos 开始的 width 位
#define SET_BITS(x, pos, width, value) \
    do { \
        unsigned int mask = ((1U << (width)) - 1) << (pos); \
        (x) = ((x) & ~mask) | (((value) << (pos)) & mask); \
    } while(0)
// 使用
unsigned int data = 0xFF;
unsigned int low_nibble = GET_BITS(data, 0, 4);    // 0x0F
unsigned int high_nibble = GET_BITS(data, 4, 4);   // 0x0F
SET_BITS(data, 4, 4, 0x0A);    // data = 0xAF
```

## 位域大小计算

**使用 sizeof 获取位域结构体大小**
`sizeof(struct <Name>)`
```c
struct PackedBits {
    unsigned int a : 1;
    unsigned int b : 1;
    unsigned int c : 1;
};
printf("Size: %zu\n", sizeof(struct PackedBits));    // 通常为 4 字节
```

---

**位域对齐与填充**
`struct <Name> { <type> <member> : <width>; ... };`
```c
struct AlignedBits {
    unsigned int a : 4;
    unsigned int : 0;    // 强制下一个成员从下一个存储单元开始
    unsigned int b : 4;
};
struct PackedBits {
    unsigned int a : 4;
    unsigned int b : 4;    // 紧跟在 a 后面
};
```
