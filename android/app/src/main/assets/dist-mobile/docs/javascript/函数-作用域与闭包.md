# JavaScript 函数、作用域与闭包

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 函数定义

**函数声明**：存在函数提升
`function <函数名>(<参数>) { ... return <值>; }`
```javascript
// 可以在声明前调用
console.log(add(2, 3));  // 5
function add(a, b) {
  return a + b;
}
```

---

**函数表达式**：不存在函数提升
`const <变量名> = function(<参数>) { ... };`
```javascript
const subtract = function (a, b) {
  return a - b;
};
console.log(subtract(5, 2));  // 3
```

---

**箭头函数**：ES6+，语法简洁
`const <变量名> = (<参数>) => <表达式>;`
```javascript
// 基本语法
const add = (a, b) => {
  return a + b;
};

// 单个参数可以省略括号
const double = num => num * 2;

// 单个表达式可以省略花括号和 return
const triple = num => num * 3;

// 无参数
const greet = () => console.log('Hello!');

// 多个参数
const multiply = (a, b) => a * b;
```

---

## 函数参数

**默认参数**：ES6+，参数默认值
`function <函数名>(<参数> = <默认值>) { ... }`
```javascript
function greet(name = '世界') {
  return `Hello, ${name}!`;
}
console.log(greet());        // 'Hello, 世界!'
console.log(greet('张三'));  // 'Hello, 张三!'
```

---

**剩余参数**：ES6+，收集剩余参数
`function <函数名>(...<参数名>) { ... }`
```javascript
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3, 4, 5));  // 15
```

---

**解构参数**：ES6+，解构对象参数
`function <函数名>({ <属性1>, <属性2> }) { ... }`
```javascript
function printUser({ name, age }) {
  console.log(`Name: ${name}, Age: ${age}`);
}
const user = { name: '张三', age: 30 };
printUser(user);  // Name: 张三, Age: 30
```

## 作用域

**全局作用域**：在所有函数外部定义
`const <变量名> = <值>;`
```javascript
const globalVar = '全局变量';
function test() {
  console.log(globalVar);  // 可以访问全局变量
}
test();
```

---

**函数作用域**：在函数内部定义
`function <函数名>() { const <变量名> = <值>; }`
```javascript
function test() {
  const localVar = '局部变量';
  console.log(localVar);  // 可以访问局部变量
}
test();
// console.log(localVar);  // 错误: localVar is not defined
```

---

**块级作用域**：ES6+，使用 let 或 const
`{ let <变量名> = <值>; }`
```javascript
if (true) {
  let blockVar = '块级变量';
  const constBlockVar = '常量块级变量';
  console.log(blockVar);  // 可以访问
}
// console.log(blockVar);  // 错误: blockVar is not defined
```

---

**作用域链**：从内到外查找变量
`function <外层>() { const <变量> = <值>; function <内层>() { console.log(<变量>); } }`
```javascript
const globalVar = '全局变量';
function outer() {
  const outerVar = '外部函数变量';
  function inner() {
    const innerVar = '内部函数变量';
    console.log(innerVar);   // 内部函数变量
    console.log(outerVar);   // 外部函数变量
    console.log(globalVar);  // 全局变量
  }
  inner();
}
outer();
```

## 闭包

**闭包基本结构**：函数访问外部作用域变量
`function <外层函数>() { let <变量>; return function() { return ++<变量>; }; }`
```javascript
function createCounter() {
  let count = 0;  // 私有变量
  return function () {
    return ++count;
  };
}
const counter = createCounter();
console.log(counter());  // 1
console.log(counter());  // 2
console.log(counter());  // 3
```

---

**闭包计数器**：封装私有变量
`function createCounter(<初始值>) { let count = <初始值>; return { ... }; }`
```javascript
function createCounter(initialValue = 0) {
  let count = initialValue;
  return {
    increment: function () { return ++count; },
    decrement: function () { return --count; },
    reset: function () { count = initialValue; return count; },
    getCount: function () { return count; },
  };
}
const counter = createCounter(10);
console.log(counter.increment());  // 11
console.log(counter.decrement());  // 10
console.log(counter.getCount());   // 10
```

---

**模块模式**：使用立即执行函数
`const <模块名> = (function() { ... return { ... }; })();`
```javascript
const mathModule = (function () {
  const PI = 3.14159;  // 私有变量
  function add(a, b) { return a + b; }  // 私有函数
  return {
    add: add,
    getPI: function () { return PI; },
  };
})();
console.log(mathModule.add(5, 3));   // 8
console.log(mathModule.getPI());      // 3.14159
```

---

**循环中的闭包问题**：使用 let 解决
`for (let i = 0; ...) { setTimeout(() => console.log(i), ...); }`
```javascript
// 问题代码：使用 var，所有输出都是 5
for (var i = 0; i < 5; i++) {
  setTimeout(function () {
    console.log(i);
  }, 1000);
}

// 解决方案 1：使用 let
for (let i = 0; i < 5; i++) {
  setTimeout(function () {
    console.log(i);  // 0, 1, 2, 3, 4
  }, 1000);
}

// 解决方案 2：使用 IIFE
for (var i = 0; i < 5; i++) {
  (function (j) {
    setTimeout(function () {
      console.log(j);  // 0, 1, 2, 3, 4
    }, 1000);
  })(i);
}
```

## this 指向

**普通函数中的 this**：指向调用者
`<对象>.<方法>();`
```javascript
function test() {
  console.log(this);
}

// 直接调用，this 指向全局对象
test();

// 作为对象方法调用，this 指向对象
const obj = { name: '张三', test: test };
obj.test();  // this 指向 obj
```

---

**箭头函数中的 this**：继承父级作用域
`const <方法> = () => { console.log(this); };`
```javascript
const obj = {
  name: '张三',
  regularFunction: function () {
    console.log(this.name);  // '张三'
    // 普通函数，this 指向全局对象
    setTimeout(function () {
      console.log(this.name);  // undefined
    }, 1000);
    // 箭头函数，继承父级作用域的 this
    setTimeout(() => {
      console.log(this.name);  // '张三'
    }, 1000);
  },
};
obj.regularFunction();
```

---

**构造函数中的 this**：指向新创建的实例
`function <构造函数>(<参数>) { this.<属性> = <参数>; }`
```javascript
function Person(name, age) {
  this.name = name;
  this.age = age;
  this.greet = function () {
    console.log(`Hello, my name is ${this.name}`);
  };
}
const person = new Person('张三', 30);
person.greet();  // Hello, my name is 张三
```

## 显式绑定 this

**call() 方法**：直接调用函数并指定 this
`<函数>.call(<thisArg>, <参数1>, <参数2>, ...)`
```javascript
function greet(greeting) {
  console.log(`${greeting}, ${this.name}!`);
}
const person = { name: '张三' };
greet.call(person, 'Hello');  // Hello, 张三!
```

---

**apply() 方法**：以数组形式传递参数
`<函数>.apply(<thisArg>, [<参数数组>])`
```javascript
function sum(a, b, c) {
  return a + b + c;
}
const numbers = [1, 2, 3];
const result = sum.apply(null, numbers);
console.log(result);  // 6
```

---

**bind() 方法**：创建绑定 this 的新函数
`const <新函数> = <函数>.bind(<thisArg>);`
```javascript
function greet() {
  console.log(`Hello, ${this.name}!`);
}
const person = { name: '张三' };
const boundGreet = greet.bind(person);
boundGreet();  // Hello, 张三!
```

## 高阶函数

**防抖函数**：延迟执行
`function debounce(<函数>, <延迟>) { ... }`
```javascript
function debounce(func, delay) {
  let timeoutId;
  return function (...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => {
      func.apply(this, args);
    }, delay);
  };
}

const debouncedSearch = debounce(function (query) {
  console.log('搜索:', query);
}, 300);
```

---

**节流函数**：限制执行频率
`function throttle(<函数>, <间隔>) { ... }`
```javascript
function throttle(func, limit) {
  let inThrottle;
  return function (...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => {
        inThrottle = false;
      }, limit);
    }
  };
}

const throttledScroll = throttle(function () {
  console.log('滚动事件触发');
}, 1000);
window.addEventListener('scroll', throttledScroll);
```
