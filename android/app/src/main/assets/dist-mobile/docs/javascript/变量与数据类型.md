# JavaScript 变量与数据类型

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 原始类型

**number**：数值类型
`let <变量名> = <数值>;`
```javascript
let integer = 42;           // 整数
let float = 3.14;            // 浮点数
let sciNotation = 1.23e5;    // 科学记数法
let notANumber = NaN;        // 特殊值
let positiveInfinity = Infinity;
```

---

**string**：字符串类型
`let <变量名> = '<字符串>';`
```javascript
let singleQuotes = 'Hello';
let doubleQuotes = "World";
let templateString = `Hello, ${doubleQuotes}!`;  // 模板字符串

console.log(templateString.length);  // 字符串长度
console.log(templateString[0]);      // 字符串索引
```

---

**boolean**：布尔类型
`let <变量名> = <true|false>;`
```javascript
let isTrue = true;
let isFalse = false;

// 布尔转换
console.log(Boolean(0));        // false
console.log(Boolean(1));        // true
console.log(Boolean(''));       // false
console.log(Boolean('Hello')); // true
console.log(Boolean(null));     // false
```

---

**undefined**：未定义类型
`let <变量名>;`
```javascript
let undefinedVar;
console.log(undefinedVar);  // 输出: undefined

// 函数默认返回值
function noReturn() {}
console.log(noReturn());  // 输出: undefined
```

---

**null**：空值类型
`let <变量名> = null;`
```javascript
let emptyObject = null;
console.log(emptyObject);  // 输出: null
console.log(typeof null);  // 输出: object (历史遗留问题)
```

---

**symbol**：符号类型（ES6+）
`let <变量名> = Symbol([<描述>]);`
```javascript
let sym1 = Symbol();
let sym2 = Symbol('description');
let sym3 = Symbol('description');

console.log(sym2 === sym3);  // false，每个 symbol 都是唯一的

// 用作对象属性
const obj = {
  [sym1]: 'value1',
};
```

---

**bigint**：大整数类型（ES2020+）
`let <变量名> = <整数>n;`
```javascript
let bigInt1 = 9007199254740991n;
let bigInt2 = BigInt(9007199254740991);

console.log(bigInt1 + 1n);  // 9007199254740992n
// console.log(bigInt1 + 1);  // 报错: 不能与 Number 直接运算
```

---

## 引用类型

**Object**：对象类型
`let <变量名> = { <键>: <值> };`
```javascript
let obj1 = {
  name: 'Alice',
  age: 30,
  address: {
    city: 'New York',
  },
};

// 构造函数创建
let obj2 = new Object();
obj2.name = 'Bob';
```

---

**Array**：数组类型
`let <变量名> = [ <元素1>, <元素2>, ... ];`
```javascript
let arr1 = [1, 'Hello', true, null];
let arr2 = new Array(5);        // 创建长度为 5 的空数组
let arr3 = new Array(1, 2, 3);  // 创建包含元素的数组

console.log(arr1.length);  // 4
```

---

**Function**：函数类型
`function <函数名>(<参数>) { ... }`
```javascript
// 函数声明
function add(a, b) {
  return a + b;
}

// 函数表达式
const multiply = function (a, b) {
  return a * b;
};

// 箭头函数
const divide = (a, b) => a / b;
```

## 类型检测

**typeof 运算符**：检测变量类型
`typeof <变量>`
```javascript
console.log(typeof 42);           // "number"
console.log(typeof 'Hello');     // "string"
console.log(typeof true);         // "boolean"
console.log(typeof undefined);   // "undefined"
console.log(typeof null);         // "object" (历史遗留问题)
console.log(typeof Symbol());     // "symbol"
console.log(typeof 1n);           // "bigint"
console.log(typeof {});           // "object"
console.log(typeof []);           // "object"
console.log(typeof function(){}); // "function"
```

---

**instanceof 运算符**：检测对象实例
`<对象> instanceof <构造函数>`
```javascript
let arr = [];
let obj = {};

console.log(arr instanceof Array);    // true
console.log(arr instanceof Object);   // true
console.log(obj instanceof Object);   // true

// 不能用于原始类型
console.log(42 instanceof Number);   // false
```

---

**Array.isArray()**：检测数组
`Array.isArray(<变量>)`
```javascript
console.log(Array.isArray([]));       // true
console.log(Array.isArray({}));       // false
console.log(Array.isArray('Hello'));   // false
```

---

**Object.prototype.toString.call()**：精确检测类型
`Object.prototype.toString.call(<变量>)`
```javascript
function getType(value) {
  return Object.prototype.toString.call(value).slice(8, -1);
}

console.log(getType(42));        // "Number"
console.log(getType('Hello'));  // "String"
console.log(getType(null));      // "Null"
console.log(getType([]));        // "Array"
console.log(getType(new Date())); // "Date"
```

## 类型转换

**转换为数字**：使用 Number() 或 parseInt()
`Number(<值>)` | `parseInt(<字符串>, <基数>)`
```javascript
// Number() 函数
console.log(Number('10'));      // 10
console.log(Number('10.5'));    // 10.5
console.log(Number(''));        // 0
console.log(Number('Hello'));   // NaN
console.log(Number(true));      // 1
console.log(Number(null));      // 0
console.log(Number(undefined)); // NaN

// parseInt() 函数
console.log(parseInt('10'));      // 10
console.log(parseInt('10.5'));    // 10
console.log(parseInt('10px'));    // 10
console.log(parseInt('10', 10));  // 10（指定基数）
```

---

**转换为字符串**：使用 String() 或 toString()
`String(<值>)` | `<值>.toString()`
```javascript
// String() 函数
console.log(String(10));         // "10"
console.log(String(true));       // "true"
console.log(String(null));        // "null"
console.log(String(undefined));   // "undefined"

// toString() 方法
console.log((10).toString());     // "10"
console.log((10.5).toString());   // "10.5"
console.log(true.toString());     // "true"
// null 和 undefined 没有 toString() 方法
```

---

**转换为布尔值**：使用 Boolean() 或 !!
`Boolean(<值>)` | `!!<值>`
```javascript
// Boolean() 函数
console.log(Boolean(1));        // true
console.log(Boolean(0));        // false
console.log(Boolean('Hello'));   // true
console.log(Boolean(''));        // false
console.log(Boolean(null));      // false
console.log(Boolean(undefined)); // false
console.log(Boolean(NaN));       // false
console.log(Boolean({}));        // true
console.log(Boolean([]));        // true

// 双重否定运算符
console.log(!!1);        // true
console.log(!!0);        // false
console.log(!!'Hello');  // true
```

---

## 隐式类型转换

**字符串拼接**：数字转换为字符串
`<数字> + <字符串>`
```javascript
console.log(1 + '2');        // "12"
console.log('Hello' + 1);   // "Hello1"
console.log(true + ' world'); // "true world"
```

---

**数学运算**：字符串转换为数字
`<字符串> - <数字>` | `+<值>`
```javascript
console.log('10' - 5);    // 5
console.log('10' * 2);    // 20
console.log('10' / 2);    // 5
console.log(+'10');       // 10
console.log(+true);       // 1
console.log(+false);      // 0
console.log(+null);       // 0
console.log(+undefined);  // NaN
```

---

**比较运算**：相等比较的隐式转换
`<值1> == <值2>`
```javascript
console.log(1 == '1');           // true（字符串转换为数字）
console.log(0 == false);         // true（布尔值转换为数字）
console.log(null == undefined);  // true（特殊情况）
console.log('' == false);        // true
```
