# JavaScript 数据类型与运算符

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 算术运算符

**基本算术运算**：执行数学运算
`<操作数1> <运算符> <操作数2>`
```javascript
console.log(1 + 2);   // 3 加法
console.log(5 - 3);    // 2 减法
console.log(2 * 3);    // 6 乘法
console.log(10 / 2);   // 5 除法
console.log(10 % 3);   // 1 取余
console.log(2 ** 3);   // 8 幂运算
```

---

**自增自减**：变量值加 1 或减 1
`<变量>++` | `++<变量>` | `<变量>--` | `--<变量>`
```javascript
let a = 1;
console.log(a++);  // 1 先使用后自增
console.log(a);     // 2

let b = 1;
console.log(++b);  // 2 先自增后使用
```

---

**字符串连接**：使用 + 运算符
`<字符串> + <其他类型>`
```javascript
console.log('Hello' + ' ' + 'World');  // "Hello World"
console.log(1 + '2');                   // "12"（隐式类型转换）
```

---

## 比较运算符

**相等比较**：== 与 ===
`<值1> == <值2>` | `<值1> === <值2>`
```javascript
// 相等（强制类型转换）
console.log(1 == '1');           // true
console.log(0 == false);         // true
console.log(null == undefined);  // true

// 严格相等（不强制类型转换）
console.log(1 === '1');           // false
console.log(0 === false);         // false
console.log(null === undefined);  // false
```

---

**大小比较**：比较两个值
`<值1> < <运算符> <值2>`
```javascript
console.log(5 > 3);    // true
console.log(5 < 3);    // false
console.log(5 >= 5);   // true
console.log(5 <= 3);   // false

// 字符串比较（按字典序）
console.log('apple' > 'banana');   // false
console.log('apple' < 'banana');   // true
```

---

## 逻辑运算符

**逻辑与**：短路求值
`<表达式1> && <表达式2>`
```javascript
console.log(true && false);   // false
console.log(true && true);    // true
console.log(false && true);   // false（短路）

// 短路求值返回操作数
console.log('hello' && 'world');  // 'world'
console.log('' && 'world');        // ''
```

---

**逻辑或**：短路求值
`<表达式1> || <表达式2>`
```javascript
console.log(true || false);    // true（短路）
console.log(false || true);    // true
console.log(false || false);   // false

// 短路求值返回操作数
console.log('hello' || 'world');  // 'hello'
console.log('' || 'world');        // 'world'
```

---

**逻辑非**：取反
`!<表达式>`
```javascript
console.log(!true);    // false
console.log(!false);   // true
console.log(!0);       // true
console.log(!'');      // true
console.log(!null);    // true
console.log(!'Hello'); // false
```

---

**空值合并**：ES2020+，仅在 null 或 undefined 时取默认值
`<值1> ?? <值2>`
```javascript
console.log(null ?? 'default');       // 'default'
console.log(undefined ?? 'default'); // 'default'
console.log(false ?? 'default');     // false（不是 null 或 undefined）
console.log(0 ?? 'default');          // 0
console.log('' ?? 'default');        // ''
```

---

## 三元运算符

**三元运算符**：条件表达式
`<条件> ? <表达式1> : <表达式2>`
```javascript
const age = 18;
const status = age >= 18 ? '成年人' : '未成年人';
console.log(status);  // '成年人'

// 嵌套三元运算符
const score = 85;
const grade = score >= 90 ? 'A' : score >= 80 ? 'B' : 'C';
console.log(grade);  // 'B'
```

---

## 赋值运算符

**基本赋值**：给变量赋值
`<变量> = <值>`
```javascript
let a = 10;
a = 20;
```

---

**复合赋值**：运算并赋值
`<变量> <运算符>= <值>`
```javascript
let a = 10;
a += 5;   // a = a + 5 = 15
a -= 3;   // a = a - 3 = 12
a *= 2;   // a = a * 2 = 24
a /= 4;   // a = a / 4 = 6
a %= 4;   // a = a % 4 = 2
a **= 3;  // a = a ** 3 = 8
```

---

**逻辑赋值**：ES2021+
`<变量> <&&=|| ||=|??=> <值>`
```javascript
let a = null;
a ||= 'default';  // a = 'default'
console.log(a);    // 'default'

let b = null;
b ??= 'default';  // b = 'default'
console.log(b);    // 'default'

let c = 0;
c ??= 'default';  // c = 0（不是 null 或 undefined）
console.log(c);    // 0
```

---

## 展开运算符

**展开数组**：展开数组元素
`...<数组>`
```javascript
let arr1 = [1, 2, 3];
let arr2 = [4, 5, 6];
let combined = [...arr1, ...arr2];
console.log(combined);  // [1, 2, 3, 4, 5, 6]

// 函数参数
function sum(...numbers) {
  return numbers.reduce((acc, curr) => acc + curr, 0);
}
console.log(sum(1, 2, 3, 4, 5));  // 15
```

---

**展开对象**：展开对象属性
`{ ...<对象> }`
```javascript
let obj1 = { a: 1, b: 2 };
let obj2 = { c: 3, d: 4 };
let merged = { ...obj1, ...obj2 };
console.log(merged);  // { a: 1, b: 2, c: 3, d: 4 }
```

---

## 可选链运算符

**可选链**：安全访问嵌套属性
`<对象>?.<属性>`
```javascript
let user = {
  name: 'Alice',
  address: {
    street: '123 Main St',
  },
};

console.log(user.address?.city);     // undefined
console.log(user.address?.street);    // '123 Main St'
console.log(user.contact?.phone);    // undefined
```

## 运算符优先级

**优先级示例**：使用括号明确运算顺序
`(<表达式>)`
```javascript
console.log(2 + 3 * 4);          // 14（乘法优先于加法）
console.log((2 + 3) * 4);        // 20（括号优先级最高）
console.log(2 ** 3 * 4);         // 32（幂运算优先于乘法）
console.log(true || false && false);  // true（逻辑与优先于逻辑或）
```
