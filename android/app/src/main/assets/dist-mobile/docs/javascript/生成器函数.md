# JavaScript 生成器函数

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 生成器函数定义

**生成器函数声明**：使用 function* 关键字
`function* <生成器名>(<参数>) { yield <值>; }`
```javascript
function* simpleGenerator() {
  yield 1;
  yield 2;
  return 3;
}

const gen = simpleGenerator();
console.log(gen.next());  // { value: 1, done: false }
console.log(gen.next());  // { value: 2, done: false }
console.log(gen.next());  // { value: 3, done: true }
```

---

**生成器函数表达式**：使用 function* 赋值
`const <变量名> = function*(<参数>) { ... };`
```javascript
const generator = function* () {
  yield 'a';
  yield 'b';
};

const gen = generator();
console.log(gen.next().value);  // 'a'
console.log(gen.next().value);  // 'b'
```

## yield 表达式

**yield 基本用法**：暂停生成器并返回值
`yield <值>;`
```javascript
function* numberGenerator() {
  yield 1;
  yield 2;
  yield 3;
}

const gen = numberGenerator();
console.log(gen.next());  // { value: 1, done: false }
console.log(gen.next());  // { value: 2, done: false }
console.log(gen.next());  // { value: 3, done: false }
console.log(gen.next());  // { value: undefined, done: true }
```

---

**yield 接收值**：next() 传递参数
`const <变量> = yield <值>;`
```javascript
function* dialog() {
  const name = yield '你叫什么名字？';
  const age = yield `你好 ${name}，你多大了？`;
  return `${name}，${age}岁`;
}

const it = dialog();
console.log(it.next());          // { value: '你叫什么名字？', done: false }
console.log(it.next('Alice'));  // { value: '你好 Alice，你多大了？', done: false }
console.log(it.next(25));        // { value: 'Alice，25岁', done: true }
```

---

**yield* 委托**：委托给另一个生成器
`yield* <可迭代对象>;`
```javascript
function* inner() {
  yield 'a';
  yield 'b';
}

function* outer() {
  yield 1;
  yield* inner();  // 委托给 inner 生成器
  yield 2;
}

for (const value of outer()) {
  console.log(value);  // 1, 'a', 'b', 2
}
```

---

**yield* 委托数组**：委托给数组
`yield* <数组>;`
```javascript
function* arrayGenerator() {
  yield* [1, 2, 3];
  yield* ['a', 'b', 'c'];
}

for (const value of arrayGenerator()) {
  console.log(value);  // 1, 2, 3, 'a', 'b', 'c'
}
```

## 生成器方法

**next()**：恢复执行
`<生成器>.next([<值>])`
```javascript
function* counter() {
  let count = 0;
  while (true) {
    const reset = yield count++;
    if (reset) {
      count = 0;
    }
  }
}

const gen = counter();
console.log(gen.next().value);  // 0
console.log(gen.next().value);  // 1
console.log(gen.next().value);  // 2
console.log(gen.next(true).value);  // 0（重置）
```

---

**return()**：终止生成器
`<生成器>.return(<值>)`
```javascript
function* generator() {
  yield 1;
  yield 2;
  yield 3;
}

const gen = generator();
console.log(gen.next());          // { value: 1, done: false }
console.log(gen.return('done'));  // { value: 'done', done: true }
console.log(gen.next());          // { value: undefined, done: true }
```

---

**throw()**：在 yield 处抛出错误
`<生成器>.throw(<错误>)`
```javascript
function* generator() {
  try {
    yield 1;
    yield 2;
  } catch (error) {
    console.log('捕获错误:', error.message);
    yield 3;
  }
}

const gen = generator();
console.log(gen.next());                    // { value: 1, done: false }
console.log(gen.throw(new Error('出错')));  // 捕获错误: 出错 → { value: 3, done: false }
```

## 迭代协议

**自定义可迭代对象**：实现 Symbol.iterator
`[Symbol.iterator]() { return { next() { ... } }; }`
```javascript
const range = {
  from: 1,
  to: 5,
  [Symbol.iterator]() {
    let current = this.from;
    const last = this.to;
    return {
      next() {
        return current <= last
          ? { value: current++, done: false }
          : { done: true };
      },
    };
  },
};

for (const num of range) {
  console.log(num);  // 1, 2, 3, 4, 5
}
```

---

**生成器实现迭代器**：简化迭代器创建
`[Symbol.iterator]() { for (...) yield <值>; }`
```javascript
const range = {
  from: 1,
  to: 5,
  *[Symbol.iterator]() {
    for (let i = this.from; i <= this.to; i++) {
      yield i;
    }
  },
};

for (const num of range) {
  console.log(num);  // 1, 2, 3, 4, 5
}
```

## 生成器应用

**无限序列**：生成无限序列
`function* <生成器>() { while (true) yield <值>; }`
```javascript
function* naturalNumbers() {
  let num = 1;
  while (true) {
    yield num++;
  }
}

const gen = naturalNumbers();
console.log(gen.next().value);  // 1
console.log(gen.next().value);  // 2
console.log(gen.next().value);  // 3
// 可以无限调用
```

---

**斐波那契数列**：生成斐波那契数
`function* fibonacci() { ... }`
```javascript
function* fibonacci() {
  let a = 0, b = 1;
  while (true) {
    yield a;
    [a, b] = [b, a + b];
  }
}

const gen = fibonacci();
console.log(gen.next().value);  // 0
console.log(gen.next().value);  // 1
console.log(gen.next().value);  // 1
console.log(gen.next().value);  // 2
console.log(gen.next().value);  // 3
```

---

**ID 生成器**：生成唯一 ID
`function* idGenerator() { let id = 0; while (true) yield ++id; }`
```javascript
function* idGenerator() {
  let id = 0;
  while (true) {
    yield ++id;
  }
}

const gen = idGenerator();
console.log(gen.next().value);  // 1
console.log(gen.next().value);  // 2
console.log(gen.next().value);  // 3
```

## 异步生成器

**异步生成器函数**：使用 async function*
`async function* <生成器>(<参数>) { yield await <promise>; }`
```javascript
async function* fetchPages(baseUrl) {
  let page = 1;
  let hasMore = true;
  while (hasMore) {
    const response = await fetch(`${baseUrl}?page=${page}`);
    const data = await response.json();
    yield data.items;
    hasMore = data.hasNext;
    page++;
  }
}

// 消费异步生成器
async function processAll() {
  for await (const items of fetchPages('/api/posts')) {
    console.log(items);
  }
}
```

---

**for await...of**：遍历异步生成器
`for await (const <变量> of <异步生成器>) { ... }`
```javascript
async function* asyncGenerator() {
  yield await new Promise((resolve) => setTimeout(() => resolve(1), 1000));
  yield await new Promise((resolve) => setTimeout(() => resolve(2), 1000));
  yield await new Promise((resolve) => setTimeout(() => resolve(3), 1000));
}

async function consume() {
  for await (const num of asyncGenerator()) {
    console.log(num);  // 1（1秒后）, 2（2秒后）, 3（3秒后）
  }
}
consume();
```

## 生成器与 Promise

**生成器执行器**：自动执行生成器
`function run(<生成器函数>) { ... }`
```javascript
function run(generatorFunc) {
  const gen = generatorFunc();

  function handle(result) {
    if (result.done) return Promise.resolve(result.value);
    return Promise.resolve(result.value).then(
      (data) => handle(gen.next(data)),
      (error) => handle(gen.throw(error))
    );
  }

  try {
    return handle(gen.next());
  } catch (error) {
    return Promise.reject(error);
  }
}

// 使用
run(function* () {
  const user = yield fetch('/api/user').then((r) => r.json());
  const posts = yield fetch(`/api/posts?userId=${user.id}`).then((r) => r.json());
  console.log(posts);
});
```

## 生成器状态

**生成器状态检查**：通过 done 属性
`<生成器>.next().done`
```javascript
function* generator() {
  yield 1;
  yield 2;
}

const gen = generator();
console.log(gen.next());  // { value: 1, done: false }
console.log(gen.next());  // { value: 2, done: false }
console.log(gen.next());  // { value: undefined, done: true }
console.log(gen.next());  // { value: undefined, done: true }（已完成）
```

## 生成器方法对比

**方法对比表**：生成器方法说明
| 方法 | 说明 | 返回值 |
|------|------|--------|
| `next(value)` | 恢复执行 | `{ value, done }` |
| `return(value)` | 终止生成器 | `{ value, done: true }` |
| `throw(error)` | 在 yield 处抛出错误 | `{ value, done }` |
```javascript
function* gen() {
  try {
    yield 1;
    yield 2;
  } catch (e) {
    console.log('caught:', e.message);
  }
  yield 3;
}

const g = gen();
g.next();   // { value: 1, done: false }
g.throw(new Error('error'));  // caught: error → { value: 3, done: false }
g.next();   // { value: undefined, done: true }
```
