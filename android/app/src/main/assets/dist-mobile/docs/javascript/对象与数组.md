# JavaScript 对象与数组

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 对象创建

**对象字面量**：直接创建对象
`const <变量名> = { <属性>: <值>, <方法>() { ... } };`
```javascript
const person = {
  name: '张三',
  age: 30,
  greet() {
    return `Hello, my name is ${this.name}`;
  },
};
console.log(person.name);   // 张三
console.log(person.greet()); // Hello, my name is 张三
```

---

**构造函数创建**：使用 new 关键字
`function <构造函数>(<参数>) { this.<属性> = <参数>; }`
```javascript
function Car(make, model, year) {
  this.make = make;
  this.model = model;
  this.year = year;
  this.getDescription = function () {
    return `${this.year} ${this.make} ${this.model}`;
  };
}
const car = new Car('Toyota', 'Camry', 2020);
console.log(car.getDescription());  // 2020 Toyota Camry
```

---

**Object.create()**：基于原型创建对象
`const <变量名> = Object.create(<原型对象>);`
```javascript
const personPrototype = {
  greet() {
    return `Hello, my name is ${this.name}`;
  },
};
const person = Object.create(personPrototype);
person.name = '张三';
console.log(person.greet());  // Hello, my name is 张三
```

---

**ES6 类**：使用 class 关键字
`class <类名> { constructor(<参数>) { ... } <方法>() { ... } }`
```javascript
class Person {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  greet() {
    return `Hello, my name is ${this.name}`;
  }
}

class Student extends Person {
  constructor(name, age, grade) {
    super(name, age);
    this.grade = grade;
  }
  study() {
    return `${this.name} is studying in grade ${this.grade}`;
  }
}
const student = new Student('李四', 15, 9);
console.log(student.greet());  // Hello, my name is 李四
console.log(student.study());  // 李四 is studying in grade 9
```

## 对象属性操作

**访问属性**：点表示法和方括号表示法
`<对象>.<属性>` | `<对象>['<属性>']`
```javascript
const person = { name: '张三', age: 30 };
console.log(person.name);      // 点表示法
console.log(person['age']);    // 方括号表示法

// 动态属性名
const propName = 'name';
console.log(person[propName]);  // 张三
```

---

**修改和添加属性**：直接赋值
`<对象>.<属性> = <值>;`
```javascript
const person = { name: '张三' };
person.name = '王五';   // 修改属性
person.city = '上海';   // 添加新属性
```

---

**删除属性**：使用 delete
`delete <对象>.<属性>;`
```javascript
const person = { name: '张三', age: 30, city: '北京' };
delete person.city;
console.log(person);  // { name: '张三', age: 30 }
```

---

**检查属性**：使用 in 或 hasOwnProperty
`'<属性>' in <对象>` | `<对象>.hasOwnProperty('<属性>')`
```javascript
const person = { name: '张三', age: 30 };
console.log('name' in person);              // true
console.log(person.hasOwnProperty('name')); // true
console.log('toString' in person);          // true（继承的属性）
console.log(person.hasOwnProperty('toString')); // false
```

## Object 静态方法

**Object.keys()**：获取对象键数组
`Object.keys(<对象>)`
```javascript
const person = { name: '张三', age: 30, city: '北京' };
const keys = Object.keys(person);
console.log(keys);  // ['name', 'age', 'city']
```

---

**Object.values()**：获取对象值数组
`Object.values(<对象>)`
```javascript
const person = { name: '张三', age: 30, city: '北京' };
const values = Object.values(person);
console.log(values);  // ['张三', 30, '北京']
```

---

**Object.entries()**：获取键值对数组
`Object.entries(<对象>)`
```javascript
const person = { name: '张三', age: 30, city: '北京' };
const entries = Object.entries(person);
console.log(entries);  // [['name', '张三'], ['age', 30], ['city', '北京']]
```

---

**Object.assign()**：合并对象
`Object.assign(<目标对象>, <源对象1>, <源对象2>)`
```javascript
const target = { a: 1 };
const source1 = { b: 2 };
const source2 = { c: 3 };
Object.assign(target, source1, source2);
console.log(target);  // { a: 1, b: 2, c: 3 }
```

## 原型与原型链

**原型方法**：在构造函数原型上添加方法
`<构造函数>.prototype.<方法> = function() { ... };`
```javascript
function Person(name) {
  this.name = name;
}
Person.prototype.greet = function () {
  return `Hello, my name is ${this.name}`;
};
const person1 = new Person('张三');
const person2 = new Person('李四');
console.log(person1.greet());  // Hello, my name is 张三
console.log(person2.greet());  // Hello, my name is 李四
```

---

**原型链继承**：使用 Object.create
`<子类>.prototype = Object.create(<父类>.prototype);`
```javascript
function Person(name) {
  this.name = name;
}
Person.prototype.greet = function () {
  return `Hello, my name is ${this.name}`;
};

function Student(name, grade) {
  Person.call(this, name);
  this.grade = grade;
}
Student.prototype = Object.create(Person.prototype);
Student.prototype.constructor = Student;
Student.prototype.study = function () {
  return `${this.name} is studying`;
};

const student = new Student('王五', 9);
console.log(student.greet());  // Hello, my name is 王五
console.log(student.study());  // 王五 is studying
```

## 数组创建

**数组字面量**：直接创建数组
`const <变量名> = [ <元素1>, <元素2>, ... ];`
```javascript
const fruits = ['apple', 'banana', 'orange'];
```

---

**Array 构造函数**：使用 new 关键字
`new Array(<元素>)` | `new Array(<长度>)`
```javascript
const numbers = new Array(1, 2, 3, 4, 5);
const emptyArray = new Array(5);  // 创建长度为 5 的空数组
```

---

**Array.from()**：从可迭代对象创建数组
`Array.from(<可迭代对象>)`
```javascript
const str = 'hello';
const arr = Array.from(str);
console.log(arr);  // ['h', 'e', 'l', 'l', 'o']

const set = new Set([1, 2, 3]);
const arrFromSet = Array.from(set);
console.log(arrFromSet);  // [1, 2, 3]
```

---

**Array.of()**：创建包含参数的数组
`Array.of(<元素1>, <元素2>, ...)`
```javascript
const arr = Array.of(1, 2, 3, 4, 5);
console.log(arr);  // [1, 2, 3, 4, 5]
```

## 数组增删操作

**push() 和 pop()**：末尾添加和删除
`<数组>.push(<元素>)` | `<数组>.pop()`
```javascript
const fruits = ['apple', 'banana', 'orange'];
fruits.push('grape');
console.log(fruits);  // ['apple', 'banana', 'orange', 'grape']

const last = fruits.pop();
console.log(last);     // grape
console.log(fruits);  // ['apple', 'banana', 'orange']
```

---

**unshift() 和 shift()**：头部添加和删除
`<数组>.unshift(<元素>)` | `<数组>.shift()`
```javascript
const fruits = ['apple', 'banana', 'orange'];
fruits.unshift('strawberry');
console.log(fruits);  // ['strawberry', 'apple', 'banana', 'orange']

const first = fruits.shift();
console.log(first);   // strawberry
console.log(fruits);  // ['apple', 'banana', 'orange']
```

---

**splice()**：添加、删除或替换元素
`<数组>.splice(<起始索引>, <删除数量>, <新元素>)`
```javascript
const fruits = ['apple', 'banana', 'orange'];
fruits.splice(1, 1, 'pear');  // 删除索引 1 的元素，并插入 'pear'
console.log(fruits);  // ['apple', 'pear', 'orange']

fruits.splice(2, 0, 'grape');  // 在索引 2 处插入 'grape'
console.log(fruits);  // ['apple', 'pear', 'grape', 'orange']
```

## 数组访问与长度

**访问和修改元素**：通过索引
`<数组>[<索引>]`
```javascript
const fruits = ['apple', 'banana', 'orange'];
console.log(fruits[0]);  // apple
fruits[1] = 'pear';
console.log(fruits);  // ['apple', 'pear', 'orange']
```

---

**数组长度**：获取或修改长度
`<数组>.length`
```javascript
const fruits = ['apple', 'banana', 'orange'];
console.log(fruits.length);  // 3

fruits.length = 2;  // 修改长度，截断数组
console.log(fruits);  // ['apple', 'banana']
```

## 数组转换方法

**join()**：连接为字符串
`<数组>.join(<分隔符>)`
```javascript
const fruits = ['apple', 'banana', 'orange'];
const str = fruits.join(', ');
console.log(str);  // 'apple, banana, orange'
```

---

**slice()**：返回子数组
`<数组>.slice(<起始>, <结束>)`
```javascript
const numbers = [1, 2, 3, 4, 5];
const subset = numbers.slice(1, 4);
console.log(subset);   // [2, 3, 4]
console.log(numbers);  // [1, 2, 3, 4, 5]（原数组不变）
```

---

**concat()**：连接数组
`<数组>.concat(<其他数组>)`
```javascript
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = arr1.concat(arr2);
console.log(combined);  // [1, 2, 3, 4, 5, 6]
```

---

**reverse()**：反转数组
`<数组>.reverse()`
```javascript
const numbers = [1, 2, 3, 4, 5];
numbers.reverse();
console.log(numbers);  // [5, 4, 3, 2, 1]
```

---

**sort()**：排序数组
`<数组>.sort([<比较函数>])`
```javascript
const numbers = [3, 1, 4, 1, 5, 9, 2, 6];
numbers.sort((a, b) => a - b);  // 升序排序
console.log(numbers);  // [1, 1, 2, 3, 4, 5, 6, 9]

const fruits = ['banana', 'apple', 'orange'];
fruits.sort();  // 按字母顺序排序
console.log(fruits);  // ['apple', 'banana', 'orange']
```

## 数组查找方法

**find() 和 findIndex()**：查找元素
`<数组>.find(<回调函数>)` | `<数组>.findIndex(<回调函数>)`
```javascript
const numbers = [1, 2, 3, 4, 5];
const firstEven = numbers.find((num) => num % 2 === 0);
console.log(firstEven);  // 2

const firstEvenIndex = numbers.findIndex((num) => num % 2 === 0);
console.log(firstEvenIndex);  // 1
```

---

**includes()**：检查是否包含
`<数组>.includes(<元素>)`
```javascript
const fruits = ['apple', 'banana', 'orange'];
console.log(fruits.includes('apple'));  // true
console.log(fruits.includes('grape'));  // false
```

---

**indexOf() 和 lastIndexOf()**：查找索引
`<数组>.indexOf(<元素>)` | `<数组>.lastIndexOf(<元素>)`
```javascript
const numbers = [1, 2, 3, 2, 1];
console.log(numbers.indexOf(2));       // 1
console.log(numbers.lastIndexOf(2));   // 3
console.log(numbers.indexOf(5));        // -1（不存在）
```

## 解构赋值

**数组解构**：从数组提取值
`const [<变量1>, <变量2>] = <数组>;`
```javascript
const numbers = [1, 2, 3, 4, 5];
const [first, second] = numbers;
console.log(first, second);  // 1 2

// 跳过元素
const [, , third] = numbers;
console.log(third);  // 3

// 剩余元素
const [firstNum, ...rest] = numbers;
console.log(firstNum);  // 1
console.log(rest);      // [2, 3, 4, 5]

// 默认值
const [x, y, z = 10] = [1, 2];
console.log(x, y, z);  // 1 2 10
```

---

**对象解构**：从对象提取属性
`const { <属性1>, <属性2> } = <对象>;`
```javascript
const person = { name: '张三', age: 30, city: '北京' };
const { name, age } = person;
console.log(name, age);  // 张三 30

// 重命名
const { name: personName, city: personCity } = person;
console.log(personName, personCity);  // 张三 北京

// 默认值
const { name, gender = '男' } = person;
console.log(name, gender);  // 张三 男

// 剩余属性
const { name, ...rest } = person;
console.log(name);  // 张三
console.log(rest);   // { age: 30, city: '北京' }
```

---

**嵌套解构**：解构嵌套结构
`const { <属性>: { <子属性> } } = <对象>;`
```javascript
const user = {
  name: '张三',
  address: {
    city: '北京',
    district: '朝阳区',
  },
  hobbies: ['读书', '旅行', '运动'],
};

// 嵌套对象解构
const {
  name,
  address: { city },
} = user;
console.log(name, city);  // 张三 北京

// 嵌套数组解构
const {
  hobbies: [firstHobby, secondHobby],
} = user;
console.log(firstHobby, secondHobby);  // 读书 旅行
```

## 展开运算符

**数组展开**：展开数组元素
`[...<数组1>, ...<数组2>]`
```javascript
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2];
console.log(combined);  // [1, 2, 3, 4, 5, 6]

// 复制数组
const original = [1, 2, 3];
const copy = [...original];
```

---

**对象展开**：展开对象属性
`{ ...<对象1>, ...<对象2> }`
```javascript
const obj1 = { a: 1, b: 2 };
const obj2 = { b: 3, c: 4 };
const combined = { ...obj1, ...obj2 };
console.log(combined);  // { a: 1, b: 3, c: 4 }（后者覆盖前者）
```

---

**字符串展开**：展开字符串为字符数组
`[...<字符串>]`
```javascript
const str = 'hello';
const arr = [...str];
console.log(arr);  // ['h', 'e', 'l', 'l', 'o']
```
