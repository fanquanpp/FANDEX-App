# JavaScript 控制流

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## if-else 语句

**基本 if-else**：条件分支
`if (<条件>) { ... } else if (<条件>) { ... } else { ... }`
```javascript
const score = 85;
if (score >= 90) {
  console.log('优秀');
} else if (score >= 80) {
  console.log('良好');
} else if (score >= 60) {
  console.log('及格');
} else {
  console.log('不及格');
}
```

---

**三元运算符**：简单的条件赋值
`<条件> ? <表达式1> : <表达式2>`
```javascript
const age = 18;
const canVote = age >= 18 ? '可以投票' : '不能投票';
console.log(canVote);  // '可以投票'
```

---

## switch 语句

**switch 语句**：多分支选择
`switch (<表达式>) { case <值>: ... break; default: ... }`
```javascript
const day = new Date().getDay();
let dayName;
switch (day) {
  case 0:
    dayName = '星期日';
    break;
  case 1:
    dayName = '星期一';
    break;
  case 6:
    dayName = '星期六';
    break;
  default:
    dayName = '工作日';
}
console.log(dayName);
```

## for 循环

**标准 for 循环**：已知次数的循环
`for (<初始化>; <条件>; <更新>) { ... }`
```javascript
// 计算 1 到 10 的和
let sum = 0;
for (let i = 1; i <= 10; i++) {
  sum += i;
}
console.log(sum);  // 55

// 遍历数组
const fruits = ['apple', 'banana', 'orange'];
for (let i = 0; i < fruits.length; i++) {
  console.log(fruits[i]);
}
```

---

**for-in 循环**：遍历对象属性
`for (<变量> in <对象>) { ... }`
```javascript
const person = {
  name: '张三',
  age: 30,
  city: '北京',
};
for (let key in person) {
  console.log(`${key}: ${person[key]}`);
}
```

---

**for-of 循环**：遍历可迭代对象（ES6+）
`for (<变量> of <可迭代对象>) { ... }`
```javascript
// 遍历数组
const fruits = ['apple', 'banana', 'orange'];
for (let fruit of fruits) {
  console.log(fruit);
}

// 遍历字符串
const str = 'Hello';
for (let char of str) {
  console.log(char);
}

// 遍历 Map
const map = new Map([['name', '张三'], ['age', 30]]);
for (let [key, value] of map) {
  console.log(`${key}: ${value}`);
}
```

## while 循环

**while 循环**：先判断后执行
`while (<条件>) { ... }`
```javascript
let sum = 0;
let i = 1;
while (i <= 10) {
  sum += i;
  i++;
}
console.log(sum);  // 55
```

---

## do-while 循环

**do-while 循环**：先执行后判断，至少执行一次
`do { ... } while (<条件>);`
```javascript
let count = 0;
do {
  console.log(`计数: ${count}`);
  count++;
} while (count < 3);
```

## 跳转语句

**break 语句**：跳出循环或 switch
`break;`
```javascript
for (let i = 0; i < 10; i++) {
  if (i === 5) {
    break;  // 当 i 等于 5 时跳出循环
  }
  console.log(i);
}
// 输出: 0 1 2 3 4
```

---

**continue 语句**：跳过当前迭代
`continue;`
```javascript
for (let i = 0; i < 10; i++) {
  if (i % 2 !== 0) {
    continue;  // 跳过奇数
  }
  console.log(i);
}
// 输出: 0 2 4 6 8
```

---

**return 语句**：从函数返回值
`return <值>;` | `return;`
```javascript
function add(a, b) {
  return a + b;  // 返回计算结果
}

// 提前返回
function checkAge(age) {
  if (age < 18) {
    return '未成年';
  }
  return '成年';
}
```

## 异常处理

**try-catch-finally**：捕获和处理异常
`try { ... } catch (<错误>) { ... } finally { ... }`
```javascript
try {
  // 可能抛出异常的代码
  const result = JSON.parse('invalid json');
} catch (error) {
  console.error('发生错误:', error.message);
} finally {
  console.log('无论是否发生错误，都会执行');
}
```

---

**throw 语句**：抛出异常
`throw new <错误类型>(<消息>);`
```javascript
function validateAge(age) {
  if (age < 0) {
    throw new Error('年龄不能为负数');
  }
  if (age > 150) {
    throw new RangeError('年龄不能超过 150');
  }
}
```

---

**自定义异常**：继承 Error 类
`class <异常类> extends Error { constructor(<消息>) { super(<消息>); this.name = <名称>; } }`
```javascript
class ValidationError extends Error {
  constructor(message) {
    super(message);
    this.name = 'ValidationError';
  }
}

function validateEmail(email) {
  if (!email.includes('@')) {
    throw new ValidationError('邮箱格式不正确');
  }
  return true;
}

try {
  validateEmail('invalid-email');
} catch (error) {
  if (error instanceof ValidationError) {
    console.error('验证错误:', error.message);
  }
}
```

## 数组迭代方法

**forEach**：遍历数组
`<数组>.forEach(<回调函数>);`
```javascript
const fruits = ['apple', 'banana', 'orange'];
fruits.forEach((fruit, index) => {
  console.log(`${index}: ${fruit}`);
});
```

---

**map**：映射数组，返回新数组
`<数组>.map(<回调函数>);`
```javascript
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map((num) => num * 2);
console.log(doubled);  // [2, 4, 6, 8, 10]
```

---

**filter**：过滤数组，返回新数组
`<数组>.filter(<回调函数>);`
```javascript
const numbers = [1, 2, 3, 4, 5];
const evenNumbers = numbers.filter((num) => num % 2 === 0);
console.log(evenNumbers);  // [2, 4]
```

---

**reduce**：归约数组，返回单个值
`<数组>.reduce(<回调函数>, <初始值>);`
```javascript
const numbers = [1, 2, 3, 4, 5];
const sum = numbers.reduce((accumulator, currentValue) => {
  return accumulator + currentValue;
}, 0);
console.log(sum);  // 15
```
