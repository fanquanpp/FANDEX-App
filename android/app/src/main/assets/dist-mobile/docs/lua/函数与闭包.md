# Lua 函数与闭包速查

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 函数定义

**命名函数**
`function <name>(<params>) <body> end`
```lua
-- 命名函数
function add(a, b)
    return a + b;
end;
print(add(1, 2));  -- 3
```

**匿名函数赋值**
`local <name> = function(<params>) <body> end`
```lua
-- 匿名函数赋值给变量
local multiply = function(a, b)
    return a * b;
end;
print(multiply(3, 4));  -- 12
```

---

## 多返回值

**返回多个值**
`function <name>(<params>) return <v1>, <v2>, ... end`
```lua
-- 函数返回多个值
function getMinMax(a, b, c)
    local min = math.min(a, b, c);
    local max = math.max(a, b, c);
    return min, max;
end;
local minVal, maxVal = getMinMax(10, 5, 8);
print("最小值:", minVal);  -- 5
print("最大值:", maxVal);  -- 10
```

## 可变参数

**... 可变参数**
`function <name>(...) <body> end`
```lua
-- 使用 ... 表示可变参数
function sum(...)
    local total = 0;
    for i, v in ipairs({...}) do
        total = total + v;
    end;
    return total;
end;
print(sum(1, 2, 3, 4, 5));  -- 15
```

## 闭包

**闭包基础**
`function <name>() local <var> = <value> return function() <body> end end`
```lua
-- 闭包：捕获外部变量
function createCounter()
    local count = 0;
    return function()
        count = count + 1;
        return count;
    end;
end;
local counter1 = createCounter();
local counter2 = createCounter();
print(counter1());  -- 1
print(counter1());  -- 2
print(counter2());  -- 1（独立的计数器）
```

**闭包封装私有变量**
`function <name>(<args>) local <private> return {<methods>} end`
```lua
-- 闭包封装私有变量
function createPerson(name)
    local age = 0;
    return {
        setAge = function(newAge)
            age = newAge;
        end,
        getAge = function()
            return age;
        end,
        getName = function()
            return name;
        end
    };
end;
local person = createPerson("张三");
person.setAge(30);
print(person.getName(), "的年龄是", person.getAge());  -- 张三 的年龄是 30
```

## 高阶函数

**函数作为参数**
`function <name>(<func>, <value>) <body> end`
```lua
-- 接受函数作为参数
function apply(func, value)
    return func(value);
end;
-- 使用
print(apply(function(x) return x * 2 end, 5));  -- 10
```

**函数作为返回值**
`function <name>(<args>) return function(<params>) <body> end end`
```lua
-- 返回函数
function createMultiplier(factor)
    return function(n)
        return n * factor;
    end;
end;
local double = createMultiplier(2);
local triple = createMultiplier(3);
print(double(5));  -- 10
print(triple(5));  -- 15
```

## 尾调用优化

**尾递归**
`function <name>(<params>) return <name>(<args>) end`
```lua
-- 尾调用优化：最后一个动作是调用自身
function factorial(n, acc)
    acc = acc or 1;
    if n <= 1 then
        return acc;
    end;
    -- 尾调用
    return factorial(n - 1, n * acc);
end;
print(factorial(10000));  -- 不会栈溢出
```

## 事件处理器

**事件总线**
`function <name>() local <listeners> = {} return {on = ..., emit = ...} end`
```lua
-- 闭包实现事件处理器
function createEventHandler()
    local listeners = {};
    return {
        on = function(event, callback)
            if not listeners[event] then
                listeners[event] = {};
            end;
            table.insert(listeners[event], callback);
        end,
        emit = function(event, ...)
            if listeners[event] then
                for _, callback in ipairs(listeners[event]) do
                    callback(...);
                end;
            end;
        end
    };
end;
-- 使用
local eventBus = createEventHandler();
eventBus.on("userLoggedIn", function(username)
    print("用户登录:", username);
end);
eventBus.emit("userLoggedIn", "admin");  -- 用户登录: admin
```

## 函数记忆化

**memoize 缓存**
`function memoize(<func>) local <cache> = {} return function(...) <body> end end`
```lua
-- 记忆化：缓存函数结果
function memoize(func)
    local cache = {};
    return function(...)
        local key = table.concat({...}, "_");
        if not cache[key] then
            cache[key] = func(...);
        end;
        return cache[key];
    end;
end;
-- 使用
local fib = memoize(function(n)
    if n <= 1 then
        return n;
    end;
    return fib(n - 1) + fib(n - 2);
end);
print(fib(40));  -- 快速计算，因为结果被缓存
```
