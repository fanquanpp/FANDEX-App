# Lua 协程与异步速查

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 协程基础

**创建协程**
`coroutine.create(<func>)`
```lua
-- 创建协程
local co = coroutine.create(function()
    print("协程开始执行");
    coroutine.yield();  -- 挂起协程
    print("协程恢复执行");
    return "协程执行完成";
end);
print("协程状态:", coroutine.status(co));  -- suspended
```

**恢复协程**
`coroutine.resume(<co>[, <args>])`
```lua
-- 恢复协程
coroutine.resume(co);  -- 输出 协程开始执行
print("协程状态:", coroutine.status(co));  -- suspended
local success, result = coroutine.resume(co);  -- 输出 协程恢复执行
print("协程状态:", coroutine.status(co));  -- dead
print("协程返回值:", result);  -- 协程执行完成
```

## 协程状态

**四种状态**
`coroutine.status(<co>)`
```lua
-- 协程状态：suspended | running | normal | dead
local co = coroutine.create(function()
    print("协程状态:", coroutine.status(co));  -- running
end);
print("创建后:", coroutine.status(co));  -- suspended
coroutine.resume(co);  -- 输出 协程状态: running
print("执行后:", coroutine.status(co));  -- dead
```

## yield 挂起

**yield 传值**
`coroutine.yield(<value>)`
```lua
-- yield 挂起并传值
local co = coroutine.create(function()
    local value = coroutine.yield(10);  -- 挂起，返回 10
    print("协程继续", value);  -- 接收 resume 传入的值
    return 20;
end);
local status, result = coroutine.resume(co);
print("第一次 resume:", status, result);  -- true 10
status, result = coroutine.resume(co, "Hello");
print("第二次 resume:", status, result);  -- true 20
```

## 协程与迭代器

**coroutine.wrap 创建迭代器**
`coroutine.wrap(<func>)`
```lua
-- 协程创建自定义迭代器
function range(from, to)
    return coroutine.wrap(function()
        for i = from, to do
            coroutine.yield(i);
        end;
    end);
end;
-- 使用
for i in range(1, 5) do
    print(i);  -- 1, 2, 3, 4, 5
end;
```

## 生产者-消费者模式

**生产者**
`function <producer>() return coroutine.create(function() <body> end) end`
```lua
-- 生产者
function producer()
    return coroutine.create(function()
        while true do
            local value = io.read();  -- 获取输入
            coroutine.yield(value);  -- 产出值
        end;
    end);
end;
```

**消费者**
`function <consumer>(<prod>) <body> end`
```lua
-- 消费者
function consumer(prod)
    while true do
        local status, value = coroutine.resume(prod);
        if not value then break end;
        print("消费:", value);
    end;
end;
-- 使用
local prod = producer();
consumer(prod);
```

## 异步操作

**协程模拟异步**
`function <asyncFunc>(<callback>) <body> end`
```lua
-- 使用协程模拟异步操作
function asyncOperation(callback)
    print("开始异步操作");
    -- 模拟异步延迟
    local timer = 3;
    local function checkTimer()
        timer = timer - 1;
        if timer <= 0 then
            callback("异步操作完成");
        else
            print("等待中...");
            checkTimer();
        end;
    end;
    checkTimer();
end;
-- 使用协程包装
function asyncOperationWithCoroutine()
    local co = coroutine.running();
    asyncOperation(function(result)
        coroutine.resume(co, result);
    end);
    return coroutine.yield();
end;
local result = asyncOperationWithCoroutine();
print("结果:", result);  -- 结果: 异步操作完成
```

## 事件循环

**事件循环**
`function <processEvents>() <body> end`
```lua
-- 事件循环
local events = {};
function addEvent(event)
    table.insert(events, event);
end;
function processEvents()
    while #events > 0 do
        local event = table.remove(events, 1);
        event();
    end;
end;
```

**delay 函数**
`function delay(<seconds>, <callback>) <body> end`
```lua
-- 延迟执行
function delay(seconds, callback)
    local startTime = os.time();
    local function checkTime()
        if os.time() - startTime >= seconds then
            callback();
        else
            addEvent(checkTime);
        end;
    end;
    addEvent(checkTime);
end;
-- wait 函数（协程中使用）
function wait(seconds)
    local co = coroutine.running();
    delay(seconds, function()
        coroutine.resume(co);
    end);
    coroutine.yield();
end;
```

**行为树**

function behaviorTree()

            if playerInSight() then

## 网络请求处理

**协程包装网络请求**
`function httpGet(<url>) <body> return coroutine.yield() end`
```lua
-- 协程包装网络请求
function httpGet(url)
    local co = coroutine.running();
    print("发送请求到:", url);
    -- 模拟网络延迟
    local timer = 2;
    local function checkTimer()
        timer = timer - 1;
        if timer <= 0 then
            local response = '{"status": "ok", "data": "success"}';
            coroutine.resume(co, response);
        else
            print("等待响应...");
            checkTimer();
        end;
    end;
    checkTimer();
    return coroutine.yield();
end;
-- 使用
local co = coroutine.create(function()
    local response = httpGet("https://api.example.com/data");
    print("收到响应:", response);
end);
coroutine.resume(co);
```
