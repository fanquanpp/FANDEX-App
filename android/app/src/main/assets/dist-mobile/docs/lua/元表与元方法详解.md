# Lua 元表与元方法速查

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 元表基础

**setmetatable 设置元表**
`setmetatable(<table>, <metatable>)`
```lua
-- 元表基础：修改表的行为
local t = {};
local mt = {
    __index = function(table, key)
        return "default:" .. key;
    end
};
setmetatable(t, mt);
print(t.hello);  -- default:hello
```

## 常用元方法

**元方法列表**
`__index | __newindex | __call | __tostring | __add | __sub | __mul | __eq | __lt | __len | __pairs`
```lua
-- 常用元方法
-- __index：访问不存在的键时触发
-- __newindex：设置不存在的键时触发
-- __call：将表当作函数调用时触发
-- __tostring：tostring() 转换时触发
-- __add：+ 运算时触发
-- __sub：- 运算时触发
-- __mul：* 运算时触发
-- __eq：== 运算时触发
-- __lt：< 运算时触发
-- __len：# 运算时触发
-- __pairs：pairs() 迭代时触发
```

---

## __index 实现继承

**__index 元方法**
`__index = function(<table>, <key>) return <value> end`
```lua
-- __index 实现继承
local Animal = {
    speak = function(self)
        return "...";
    end
};
local Dog = setmetatable({}, {__index = Animal});
Dog.speak = function(self)
    return "Woof!";
end;
-- 实例继承
local d = setmetatable({}, {__index = Dog});
print(d:speak());  -- Woof!
```

## __call 元方法

**__call 将表当作函数调用**
`__call = function(<self>, <args>) <body> end`
```lua
-- __call 元方法
local counter = setmetatable({count = 0}, {
    __call = function(self, increment)
        self.count = self.count + (increment or 1);
        return self.count;
    end
});
-- 直接调用表
print(counter(1));  -- 1
print(counter(5));  -- 6
```

## __tostring 元方法

**__tostring 自定义字符串表示**
`__tostring = function(<self>) return <string> end`
```lua
-- __tostring 元方法
local person = setmetatable({name = "Alice", age = 30}, {
    __tostring = function(self)
        return string.format("%s (%d)", self.name, self.age);
    end
});
print(person);  -- Alice (30)
```

## __add 算术元方法

**__add 定义加法运算**
`__add = function(<a>, <b>) return <result> end`
```lua
-- __add 元方法：定义表的加法运算
local mt = {
    __add = function(a, b)
        local result = {};
        for i, v in ipairs(a) do
            result[i] = v;
        end;
        for i, v in ipairs(b) do
            result[#result + 1] = v;
        end;
        return result;
    end,
    __tostring = function(t)
        return "[" .. table.concat(t, ", ") .. "]";
    end
};
local t1 = {1, 2, 3};
local t2 = {4, 5, 6};
setmetatable(t1, mt);
setmetatable(t2, mt);
local t3 = t1 + t2;
print(t3);  -- [1, 2, 3, 4, 5, 6]
```

## __newindex 元方法

**__newindex 拦截赋值**
`__newindex = function(<table>, <key>, <value>) <body> end`
```lua
-- __newindex 元方法：拦截对不存在键的赋值
local protected = setmetatable({}, {
    __newindex = function(t, k, v)
        if k == "password" then
            error("不能修改密码字段");
        end;
        rawset(t, k, v);  -- 使用 rawset 实际设置值
    end
});
protected.name = "Alice";  -- OK
-- protected.password = "123";  -- 抛出错误
```

## __len 元方法

**__len 自定义长度**
`__len = function(<table>) return <number> end`
```lua
-- __len 元方法：自定义表的长度运算
local collection = setmetatable({items = {1, 2, 3, 4, 5}}, {
    __len = function(self)
        return #self.items;
    end
});
print(#collection);  -- 5
```

## __pairs 元方法

**__pairs 自定义遍历**
`__pairs = function(<table>) return <iterator>, <table>, <initial> end`
```lua
-- __pairs 元方法：自定义 pairs 遍历
local ordered = setmetatable({
    keys = {"a", "b", "c"},
    data = {a = 1, b = 2, c = 3}
}, {
    __pairs = function(self)
        local i = 0;
        local function iter()
            i = i + 1;
            local k = self.keys[i];
            if k then
                return k, self.data[k];
            end;
        end;
        return iter;
    end
});
for k, v in pairs(ordered) do
    print(k, v);  -- a 1, b 2, c 3
end;
```

## __eq 比较元方法

**__eq 自定义相等比较**
`__eq = function(<a>, <b>) return <bool> end`
```lua
-- __eq 元方法：自定义相等比较
local Vector = {};
Vector.__index = Vector;
Vector.__eq = function(a, b)
    return a.x == b.x and a.y == b.y;
end;
function Vector.new(x, y)
    return setmetatable({x = x, y = y}, Vector);
end;
local v1 = Vector.new(1, 2);
local v2 = Vector.new(1, 2);
local v3 = Vector.new(3, 4);
print(v1 == v2);  -- true
print(v1 == v3);  -- false
```
