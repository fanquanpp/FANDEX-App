# Lua 字符串模式匹配速查

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 模式字符类

**字符类**
`%a | %d | %l | %u | %w | %s | %p | %c | .`
```lua
-- 字符类
-- %a：任何字母
-- %d：任何数字
-- %l：任何小写字母
-- %u：任何大写字母
-- %w：任何字母和数字
-- %s：任何空白字符
-- %p：任何标点符号
-- %c：任何控制字符
-- . ：任何字符（除了换行符）
-- 大写版本表示补集：%A 非字母，%D 非数字
local s = "Hello123";
print(string.match(s, "%a+"));  -- Hello
print(string.match(s, "%d+"));  -- 123
```

**量词**
`* | + | ? | -`
```lua
-- *：0 次或多次（贪婪）
-- +：1 次或多次（贪婪）
-- ?：0 次或 1 次
-- -：0 次或多次（非贪婪）
local s = "aaa";
print(string.match(s, "a*"));  -- aaa
print(string.match(s, "a+"));  -- aaa
print(string.match(s, "a?"));  -- a
print(string.match(s, "a-"));  -- （空，匹配最短）
```

## string.find 查找

**基本查找**
`string.find(<s>, <pattern>[, <init>[, <plain>]])`
```lua
-- 返回匹配的起止位置
local s = "Hello World";
local start, finish = string.find(s, "World");
print(start, finish);  -- 7  11
-- 没找到返回 nil
local pos = string.find(s, "Lua");
print(pos);  -- nil
```

**带捕获的查找**
`string.find(<s>, "(<capture1>)(<capture2>)")`
```lua
-- 带捕获的查找
local s = "name=Alice age=25";
local start, finish, name, age = string.find(s, "name=(%w+) age=(%d+)");
print(name);  -- Alice
print(age);   -- 25
```

**纯文本查找**
`string.find(<s>, <pattern>, <init>, true)`
```lua
-- plain 参数：纯文本查找（不解释模式）
local s = "file.txt";
local pos = string.find(s, ".", 1, true);  -- true 表示纯文本查找
print(pos);  -- 5（找到字面的点号）
```

## string.match 匹配

**简单匹配**
`string.match(<s>, <pattern>)`
```lua
-- 返回匹配的内容
local s = "The price is $99.9";
local price = string.match(s, "%$(%d+%.?%d*)");
print(price);  -- 99.9
```

**多捕获**
`string.match(<s>, "(<c1>)(<c2>)(<c3>)")`
```lua
-- 多个捕获
local s = "2026-06-14";
local y, m, d = string.match(s, "(%d+)%-(%d+)%-(%d+)");
print(string.format("年: %s, 月: %s, 日: %s", y, m, d));
```

## string.gmatch 迭代匹配

**迭代匹配**
`string.gmatch(<s>, <pattern>)`
```lua
-- 提取所有数字
local s = "I have 3 apples and 5 oranges";
for num in string.gmatch(s, "%d+") do
    print(num);
end;
-- 输出: 3, 5
```

**提取键值对**
`for <k>, <v> in string.gmatch(<s>, "(<keyPat>)=(<valPat>)") do <body> end`
```lua
-- 提取所有键值对
local s = "name=Alice&age=25&city=Beijing";
local params = {};
for key, value in string.gmatch(s, "(%w+)=(%w+)") do
    params[key] = value;
end;
-- params = {name = "Alice", age = "25", city = "Beijing"}
```

## string.gsub 替换

**字符串替换**
`string.gsub(<s>, <pattern>, <repl>[, <n>])`
```lua
-- 用字符串替换
local s = "Hello World";
local result = string.gsub(s, "World", "Lua");
print(result);  -- Hello Lua
-- 限制替换次数
local s = "aaa";
local result = string.gsub(s, "a", "b", 2);  -- 只替换前 2 次
print(result);  -- bba
```

**捕获组替换**
`string.gsub(<s>, "(<c1>)%-(<c2>)", "%2/%3/%1")`
```lua
-- 用捕获组替换
local s = "2026-06-14";
local result = string.gsub(s, "(%d+)%-(%d+)%-(%d+)", "%2/%3/%1");
print(result);  -- 06/14/2026
```

**函数替换**
`string.gsub(<s>, <pattern>, <func>)`
```lua
-- 用函数替换
local s = "The temperature is 25 degrees";
local result = string.gsub(s, "%d+", function(num)
    return tostring(tonumber(num) * 9 / 5 + 32);  -- 摄氏转华氏
end);
print(result);  -- The temperature is 77 degrees
```

**表替换**
`string.gsub(<s>, <pattern>, <table>)`
```lua
-- 用表替换
local replacements = {
    hello = "hi",
    world = "earth",
};
local s = "hello world";
local result = string.gsub(s, "%w+", replacements);
print(result);  -- hi earth
```

## 锚点

**^ 和 $ 锚点**
`^<pattern> | <pattern>$`
```lua
-- ^ 匹配字符串开头
local s = "Hello World";
print(string.match(s, "^Hello"));  -- Hello
print(string.match(s, "^World"));  -- nil
-- $ 匹配字符串结尾
print(string.match(s, "World$"));  -- World
print(string.match(s, "Hello$"));  -- nil
-- 同时使用锚点检查完整匹配
local function isDigitOnly(s)
    return string.match(s, "^%d+$") ~= nil;
end;
print(isDigitOnly("12345"));  -- true
print(isDigitOnly("12a45"));  -- false
```

## 非贪婪匹配

**- 非贪婪量词**
`<pattern>.-<pattern>`
```lua
local s = "<b>bold</b> <i>italic</i>";
-- 贪婪匹配（*）：匹配最长的结果
local result = string.match(s, "<.*>");
print(result);  -- <b>bold</b> <i>italic</i>
-- 非贪婪匹配（-）：匹配最短的结果
local result = string.match(s, "<.->");
print(result);  -- <b>
-- 提取所有 HTML 标签内容
for tag in string.gmatch(s, "<(.-)>") do
    print(tag);
end;
-- 输出: b /b i /i
```

## 字符集

**方括号字符集**
`[<chars>] | [^<chars>]`
```lua
-- 匹配方括号中的任意一个字符
local s = "abc123";
print(string.match(s, "[a-z]+"));  -- abc
print(string.match(s, "[0-9]+"));  -- 123
-- 补集：以 ^ 开头
local s = "Hello 123 World";
print(string.match(s, "[^%s]+"));  -- Hello
-- 匹配十六进制字符
local hex = "1A2B3C";
for byte in string.gmatch(hex, "[%da-fA-F]+") do
    print(byte);
end;
```

## 转义特殊字符

**% 转义**
`%<char>`
```lua
-- 特殊字符需要用 % 转义
-- 错误：. 在模式中表示"任意字符"
local s = "file.txt";
local result = string.match(s, "file.txt");  -- 可能匹配 "fileXtxt"
-- 正确：用 %. 匹配字面的点号
local result = string.match(s, "file%.txt");
-- 匹配百分号本身
local s = "100%";
local result = string.match(s, "%d+%%");  -- %% 匹配字面的 %
```

## 字符串分割

**自定义 split 函数**
`function string.split(<s>, <sep>)`
```lua
-- 按分隔符分割字符串
function string.split(s, sep)
    sep = sep or "%s+";  -- 默认按空白分割
    local result = {};
    for item in string.gmatch(s, "([^" .. sep .. "]+)") do
        table.insert(result, item);
    end;
    return result;
end;
-- 使用
local parts = string.split("hello world lua", " ");
-- parts = {"hello", "world", "lua"}
local csv = string.split("a,b,c", ",");
-- csv = {"a", "b", "c"}
```

## 解析配置

**解析 key = value 配置**
`for <line> in <text>:gmatch("[^\n]+") do <body> end`
```lua
-- 解析 key = value 格式的配置
local function parseConfig(text)
    local config = {};
    for line in string.gmatch(text, "[^\n]+") do
        -- 跳过注释和空行
        if not string.match(line, "^%s*#") and not string.match(line, "^%s*$") then
            local key, value = string.match(line, "^%s*(%w+)%s*=%s*(.-)%s*$");
            if key and value then
                config[key] = value;
            end;
        end;
    end;
    return config;
end;
```

## 验证输入格式

**验证邮箱**
`string.match(<s>, "^<pattern>$")`
```lua
-- 验证邮箱格式（简化版）
local function isValidEmail(email)
    return string.match(email, "^[%w%.]+@[%w%.]+%.[%a]+$") ~= nil;
end;
print(isValidEmail("user@example.com"));  -- true
print(isValidEmail("invalid-email"));     -- false
```
