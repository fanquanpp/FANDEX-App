# 类型守卫与自定义守卫

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## typeof 类型守卫

**typeof：检查原始类型**
`if (typeof <变量> === '<类型>') { ... }`

```typescript
// typeof 类型守卫
function processValue(value: string | number | boolean) {
  if (typeof value === 'string') {
    // value 类型缩小为 string
    return value.toUpperCase();
  } else if (typeof value === 'number') {
    // value 类型缩小为 number
    return value.toFixed(2);
  } else {
    // value 类型缩小为 boolean
    return value ? 'true' : 'false';
  }
}
console.log(processValue('hello')); // HELLO
console.log(processValue(3.14159)); // 3.14
console.log(processValue(true)); // true
```

## instanceof 类型守卫

**instanceof：检查类实例**
`if (<变量> instanceof <类>) { ... }`

```typescript
class Animal {
  name: string;
  constructor(name: string) {
    this.name = name;
  }
}
class Dog extends Animal {
  bark(): void {
    console.log('Woof!');
  }
}
class Cat extends Animal {
  meow(): void {
    console.log('Meow!');
  }
}
function makeSound(animal: Animal) {
  if (animal instanceof Dog) {
    // animal 类型缩小为 Dog
    animal.bark();
  } else if (animal instanceof Cat) {
    // animal 类型缩小为 Cat
    animal.meow();
  }
}
const dog = new Dog('Buddy');
const cat = new Cat('Kitty');
makeSound(dog); // Woof!
makeSound(cat); // Meow!
```

## in 操作符类型守卫

**in：检查属性是否存在**
`if ('<属性>' in <变量>) { ... }`

```typescript
interface Fish {
  swim(): void;
}
interface Bird {
  fly(): void;
}
type Pet = Fish | Bird;
function move(pet: Pet) {
  if ('swim' in pet) {
    // pet 类型缩小为 Fish
    pet.swim();
  } else {
    // pet 类型缩小为 Bird
    pet.fly();
  }
}
```

## 自定义类型守卫

**类型谓词：使用 `is` 关键字**
`function <函数>(<参数>: <类型>): <参数> is <目标类型>`

```typescript
interface Person {
  name: string;
  age: number;
}
interface Animal {
  species: string;
  sound: string;
}
type LivingBeing = Person | Animal;
// 自定义类型守卫
function isPerson(being: LivingBeing): being is Person {
  return 'name' in being && 'age' in being;
}
function processBeing(being: LivingBeing) {
  if (isPerson(being)) {
    // being 类型缩小为 Person
    console.log(`Person: ${being.name}, ${being.age} years old`);
  } else {
    // being 类型缩小为 Animal
    console.log(`Animal: ${being.species}, makes ${being.sound}`);
  }
}
```

**自定义类型守卫：检查字符串格式**
`function <函数>(<参数>: string): <参数> is <目标类型>`

```typescript
type Email = string & { readonly __brand: unique symbol };
type URL = string & { readonly __brand: unique symbol };
// 自定义类型守卫：检查是否为邮箱
function isEmail(value: string): value is Email {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
}
// 自定义类型守卫：检查是否为 URL
function isURL(value: string): value is URL {
  try {
    new URL(value);
    return true;
  } catch {
    return false;
  }
}
function processString(value: string) {
  if (isEmail(value)) {
    console.log(`Email: ${value}`);
  } else if (isURL(value)) {
    console.log(`URL: ${value}`);
  } else {
    console.log(`Plain string: ${value}`);
  }
}
processString('alice@example.com'); // Email: alice@example.com
processString('https://example.com'); // URL: https://example.com
processString('hello'); // Plain string: hello
```

## 判别式联合类型守卫

**判别式联合：使用共同属性区分类型**
`type <类型> = { kind: '<值1>'; ... } | { kind: '<值2>'; ... }`

```typescript
interface Circle {
  kind: 'circle';
  radius: number;
}
interface Square {
  kind: 'square';
  sideLength: number;
}
interface Rectangle {
  kind: 'rectangle';
  width: number;
  height: number;
}
type Shape = Circle | Square | Rectangle;
// 使用判别式联合
function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      // shape 类型缩小为 Circle
      return Math.PI * shape.radius ** 2;
    case 'square':
      // shape 类型缩小为 Square
      return shape.sideLength ** 2;
    case 'rectangle':
      // shape 类型缩小为 Rectangle
      return shape.width * shape.height;
    default:
      // 穷尽性检查
      const exhaustiveCheck: never = shape;
      return 0;
  }
}
console.log(getArea({ kind: 'circle', radius: 5 })); // 78.53981633974483
console.log(getArea({ kind: 'square', sideLength: 4 })); // 16
console.log(getArea({ kind: 'rectangle', width: 3, height: 6 })); // 18
```

## 数组类型守卫

**Array.isArray：检查是否为数组**
`if (Array.isArray(<变量>)) { ... }`

```typescript
function process(value: string | string[]) {
  if (Array.isArray(value)) {
    // value 类型缩小为 string[]
    value.forEach(item => console.log(item));
  } else {
    // value 类型缩小为 string
    console.log(value);
  }
}
process('hello'); // hello
process(['a', 'b', 'c']); // a, b, c
```

---

**自定义数组元素类型守卫**
`function <函数>(<数组>: any[]): <数组> is <目标类型>[]`

```typescript
interface User {
  id: number;
  name: string;
}
// 自定义类型守卫：检查是否为 User 数组
function isUserArray(arr: any[]): arr is User[] {
  return arr.every(item =>
    typeof item === 'object' &&
    item !== null &&
    'id' in item &&
    'name' in item
  );
}
function processArray(arr: any[]) {
  if (isUserArray(arr)) {
    // arr 类型缩小为 User[]
    arr.forEach(user => console.log(`User: ${user.name} (ID: ${user.id})`));
  } else {
    console.log('Not a user array');
  }
}
processArray([
  { id: 1, name: 'Alice' },
  { id: 2, name: 'Bob' },
]);
// User: Alice (ID: 1)
// User: Bob (ID: 2)
```

## null 和 undefined 检查

**null 检查：使用严格相等**
`if (<变量> !== null) { ... }`

```typescript
function processValue(value: string | null) {
  if (value !== null) {
    // value 类型缩小为 string
    console.log(value.toUpperCase());
  } else {
    console.log('Value is null');
  }
}
processValue('hello'); // HELLO
processValue(null); // Value is null
```

---

**undefined 检查**
`if (<变量> !== undefined) { ... }`

```typescript
    console.log('Value is undefined');
  }
```

---

**nullish 合并运算符**
`<值> ?? <默认值>`

```typescript
function processValue(value: string | null | undefined) {
  const processed = value ?? 'default';
  console.log(processed);
}
processValue('hello'); // hello
processValue(null); // default
processValue(undefined); // default
```

---

**可选链运算符**
`<对象>?..<属性>`

```typescript
interface User {
  name: string;
  address?: {
    city: string;
    country?: string;
  };
}
function getCity(user: User): string | undefined {
  return user?.address?.city;
}
const user1: User = { name: 'Alice', address: { city: 'New York' } };
const user2: User = { name: 'Bob' };
console.log(getCity(user1)); // New York
console.log(getCity(user2)); // undefined
```

## 复杂类型守卫

**组合类型守卫**
`if (<条件1> && <条件2>) { ... }`

```typescript
interface User {
  id: number;
  name: string;
  email?: string;
  age?: number;
}
function processUser(user: User) {
  if (user.email && user.age) {
    // user.email 和 user.age 都存在
    console.log(`User: ${user.name}, Email: ${user.email}, Age: ${user.age}`);
  } else if (user.email) {
    console.log(`User: ${user.name}, Email: ${user.email}`);
  } else if (user.age) {
    console.log(`User: ${user.name}, Age: ${user.age}`);
  } else {
    console.log(`User: ${user.name}`);
  }
}
```

**类型断言函数**
`function <函数>(<参数>: unknown): asserts <参数> is <类型>`

```typescript
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== 'string') {
    throw new Error(`Expected string, got ${typeof value}`);
  }
}
function processValue(value: unknown) {
  assertIsString(value);
  // value 类型缩小为 string
  console.log(value.toUpperCase());
}
processValue('hello'); // HELLO
// processValue(42); // Error: Expected string, got number
```

---

**断言函数：检查条件**
`function <函数>(<条件>: <类型>): asserts <条件>`

```typescript
function assert(condition: unknown, message: string): asserts condition {
  if (!condition) {
    throw new Error(message);
  }
}
function processValue(value: unknown) {
  assert(typeof value === 'string', 'Value must be a string');
  // value 类型缩小为 string
  console.log(value.toUpperCase());
}
processValue('hello'); // HELLO
// processValue(42); // Error: Value must be a string
```

## 类型守卫的应用

**过滤数组**
`function <函数>(<数组>: <类型>[]): <目标类型>[]`

```typescript
interface User {
  id: number;
  name: string;
  email?: string;
}
// 过滤有 email 的用户
function filterUsersWithEmail(users: User[]): (User & { email: string })[] {
  return users.filter((user): user is User & { email: string } => user.email !== undefined);
}
const users: User[] = [
  { id: 1, name: 'Alice', email: 'alice@example.com' },
  { id: 2, name: 'Bob' },
  { id: 3, name: 'Charlie', email: 'charlie@example.com' },
];
const usersWithEmail = filterUsersWithEmail(users);
console.log(usersWithEmail);
// [{ id: 1, name: 'Alice', email: 'alice@example.com' }, { id: 3, name: 'Charlie', email: 'charlie@example.com' }]
```

**类型守卫与 switch**
`switch (<变量>.<属性>) { case '<值>': ... }`

```typescript
type RequestState =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: any }
  | { status: 'error'; error: string };
function handleRequest(state: RequestState): string {
  switch (state.status) {
    case 'idle':
      return 'Ready to start';
    case 'loading':
      return 'Loading...';
    case 'success':
      return `Data: ${JSON.stringify(state.data)}`;
    case 'error':
      return `Error: ${state.error}`;
  }
}
console.log(handleRequest({ status: 'idle' })); // Ready to start
console.log(handleRequest({ status: 'loading' })); // Loading...
console.log(handleRequest({ status: 'success', data: { id: 1 } })); // Data: {"id":1}
console.log(handleRequest({ status: 'error', error: 'Network error' })); // Error: Network error
```
