# 基础类型系统

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 原始类型

**原始类型：声明基本变量类型**
`let <变量>: <类型> = <值>`

```typescript
// 布尔值
let isDone: boolean = false;
// 数字（整数和浮点数）
let count: number = 42;
let pi: number = 3.14;
// 字符串
let name: string = "TypeScript";
let message: string = `Hello, ${name}!`;
// symbol
let sym1: symbol = Symbol("key");
// bigint
let big: bigint = 100n;
let big2: bigint = BigInt("9007199254740991");
```

---

## 数组类型

**数组声明：两种语法形式**
`let <变量>: <类型>[] = [<值>]`
`let <变量>: Array<<类型>> = [<值>]`

```typescript
// 方式 1: 类型[]
let numbers: number[] = [1, 2, 3, 4, 5];
let strings: string[] = ['a', 'b', 'c'];
// 方式 2: Array<类型>
let numbers2: Array<number> = [1, 2, 3, 4, 5];
let strings2: Array<string> = ['a', 'b', 'c'];
// 多维数组
let matrix: number[][] = [
  [1, 2],
  [3, 4],
];
```

---

## 元组类型

**元组：固定长度和类型的数组**
`let <变量>: [<类型1>, <类型2>] = [<值1>, <值2>]`

```typescript
// 基本元组
let person: [string, number] = ['John', 30];
// 访问元组元素
let name: string = person[0];
let age: number = person[1];
// 可选元素
let optionalTuple: [string, number?] = ['John'];
// 剩余元素
let restTuple: [string, ...number[]] = ['John', 1, 2, 3];
// 只读元组
let readonlyTuple: readonly [string, number] = ['John', 30];
```

## 枚举类型

**枚举：一组命名的常量**
`enum <枚举名> { <成员1>, <成员2>, ... }`

```typescript
// 基本枚举
enum Direction {
  Up,
  Down,
  Left,
  Right,
}
let dir: Direction = Direction.Up; // 0
// 自定义枚举值
enum Color {
  Red = 1,
  Green = 2,
  Blue = 4,
}
let color: Color = Color.Green; // 2
// 字符串枚举
enum Status {
  Active = 'ACTIVE',
  Inactive = 'INACTIVE',
  Pending = 'PENDING',
}
let status: Status = Status.Active; // "ACTIVE"
// 常量枚举（编译时内联）
const enum Weekday {
  Monday,
  Tuesday,
  Wednesday,
}
let day: Weekday = Weekday.Monday;
```

## 对象类型

**对象类型：内联定义结构**
`let <变量>: { <属性>: <类型> } = { <属性>: <值> }`

```typescript
// 内联对象类型
let user: { name: string; age: number } = {
  name: 'John',
  age: 30,
};
// 可选属性
let user2: { name: string; age?: number } = {
  name: 'John',
};
// 只读属性
let user3: { readonly name: string; age: number } = {
  name: 'John',
  age: 30,
};
// 索引签名
let map: { [key: string]: number } = {
  a: 1,
  b: 2,
};
map.c = 3; // 允许添加新属性
```

## any 类型

**any 类型：绕过类型检查**
`let <变量>: any = <值>`

```typescript
// 任何值都可以赋值给 any 类型
let anyValue: any = 42;
anyValue = 'Hello';
anyValue = true;
// any 类型的变量可以访问任何属性或方法
let anyObj: any = { name: 'John' };
console.log(anyObj.name); // 没问题
console.log(anyObj.age); // 没问题，运行时会是 undefined
anyObj.method(); // 没问题，运行时会报错
```

---

## unknown 类型

**unknown 类型：安全的 any，使用前必须类型缩小**
`let <变量>: unknown = <值>`

```typescript
// 任何值都可以赋值给 unknown 类型
let unknownValue: unknown = 42;
unknownValue = 'Hello';
unknownValue = true;
// unknown 类型的变量不能直接访问属性或方法
let unknownObj: unknown = { name: 'John' };
// 需要进行类型缩小
if (typeof unknownObj === 'object' && unknownObj !== null) {
  console.log((unknownObj as { name: string }).name);
}
// 或使用类型守卫
function isPerson(obj: unknown): obj is { name: string; age: number } {
  return typeof obj === 'object' && obj !== null && 'name' in obj && 'age' in obj;
}
if (isPerson(unknownObj)) {
  console.log(unknownObj.name);
  console.log(unknownObj.age);
}
```

## void 类型

**void 类型：函数无返回值**
`function <函数>(<参数>): void { ... }`

```typescript
// 无返回值的函数
function logMessage(message: string): void {
  console.log(message);
}
// 可以返回 undefined
function returnUndefined(): void {
  return undefined;
}
// void 类型的变量只能赋值 undefined 或 null
let voidVar: void = undefined;
```

## never 类型

**never 类型：永远不会有值的类型**
`function <函数>(<参数>): never { ... }`

```typescript
// 抛出异常的函数
function throwError(message: string): never {
  throw new Error(message);
}
// 无限循环的函数
function infiniteLoop(): never {
  while (true) {
    // 无限循环
  }
}
// never 类型可以赋值给任何类型
let num: number = throwError('Error');
let str: string = throwError('Error');
// 没有类型可以赋值给 never 类型（除了 never 本身）
let neverVar: never = throwError('Error');
```

## null 和 undefined 类型

**null 和 undefined：基本空值类型**
`let <变量>: null = null`
`let <变量>: undefined = undefined`

```typescript
// null 类型
let nullValue: null = null;
// undefined 类型
let undefinedValue: undefined = undefined;
// 联合类型处理空值
let numWithNull: number | null = null;
let numWithUndefined: number | undefined = undefined;
let numWithBoth: number | null | undefined = 42;
// 可选参数自动包含 undefined
function greet(name?: string) {
  console.log(`Hello, ${name || 'Guest'}!`);
}
```

## 联合类型

**联合类型：使用 `|` 表示值可以是其中之一**
`let <变量>: <类型1> | <类型2>`

```typescript
// 基本联合类型
let id: string | number;
id = '123';
id = 456;
// 联合类型的类型缩小
function processId(id: string | number) {
  if (typeof id === 'string') {
    console.log(`String ID: ${id.toUpperCase()}`);
  } else {
    console.log(`Number ID: ${id.toFixed(2)}`);
  }
}
// 联合类型与字面量类型
type Status = 'active' | 'inactive' | 'pending';
let userStatus: Status = 'active';
```

## 交叉类型

**交叉类型：使用 `&` 表示同时满足所有类型**
`type <类型> = <类型1> & <类型2>`

```typescript
// 基本交叉类型
interface Person {
  name: string;
  age: number;
}
interface Serializable {
  serialize: () => string;
}
type SerializablePerson = Person & Serializable;
let person: SerializablePerson = {
  name: "John",
  age: 30,
  serialize: function() {
    return JSON.stringify(this);
  }
};
// 交叉类型与联合类型组合
interface X {
  x: number;
}
interface Y {
  y: string;
}
type XY = X & Y;
let xy: XY = {
  x: 1,
  y: "hello"
};
```

## 类型别名

**type 别名：为类型创建新名称**
`type <别名> = <类型>`

```typescript
// 基本类型别名
type ID = string | number;
let userId: ID = '123';
let productId: ID = 456;
// 对象类型别名
type User = {
  id: ID;
  name: string;
  email: string;
  age?: number;
};
let user: User = {
  id: '123',
  name: 'John',
  email: 'john@example.com',
};
// 函数类型别名
type AddFunction = (a: number, b: number) => number;
const add: AddFunction = (a, b) => a + b;
// 泛型类型别名
type Container<T> = {
  value: T;
  getValue: () => T;
};
let numberContainer: Container<number> = {
  value: 42,
  getValue: function () {
    return this.value;
  },
};
// 递归类型别名
type TreeNode<T> = {
  value: T;
  children: TreeNode<T>[];
};
```

## 字面量类型

**字面量类型：表示具体的值**
`type <类型> = <值1> | <值2>`

```typescript
// 字符串字面量类型
type Direction = 'North' | 'South' | 'East' | 'West';
let move: Direction = 'North';
// 数字字面量类型
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;
let roll: DiceRoll = 4;
// 布尔字面量类型
type OnlyTrue = true;
let isActive: OnlyTrue = true;
// 字面量类型与类型守卫
type Shape =
  | { kind: 'circle'; radius: number }
  | { kind: 'square'; sideLength: number }
  | { kind: 'rectangle'; width: number; height: number };
function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    case 'rectangle':
      return shape.width * shape.height;
    default:
      return 0;
  }
}
```

## 类型断言

**类型断言：告诉编译器变量的实际类型**
`<值> as <类型>`
`<<类型>><值>`

```typescript
// as 语法（推荐）
let someValue: any = 'this is a string';
let strLength: number = (someValue as string).length;
// 尖括号语法
let someValue2: any = 'this is a string';
let strLength2: number = (<string>someValue2).length;
// 双重断言
let value: unknown = 'hello';
let str: string = value as any as string;
// 非空断言（使用 ! 操作符）
function getElement(id: string): HTMLElement | null {
  return document.getElementById(id);
}
let element = getElement('myElement')!;
console.log(element.textContent);
// as const 断言
const config = {
  apiUrl: 'https://api.example.com',
  timeout: 5000,
} as const;
// 数组 as const
const numbers = [1, 2, 3] as const;
// numbers 类型为 readonly [1, 2, 3]
```

## 类型守卫

**typeof 类型守卫：检查原始类型**
`if (typeof <变量> === '<类型>') { ... }`

```typescript
function processValue(value: string | number) {
  if (typeof value === 'string') {
    // value 类型缩小为 string
    console.log(value.toUpperCase());
  } else {
    // value 类型缩小为 number
    console.log(value.toFixed(2));
  }
}
```

---

**instanceof 类型守卫：检查类实例**
`if (<变量> instanceof <类>) { ... }`

```typescript
class Animal {
  name: string;
  constructor(name: string) {
    this.name = name;
  }
}
class Dog extends Animal {
  bark() {
    console.log('Woof!');
  }
}
class Cat extends Animal {
  meow() {
    console.log('Meow!');
  }
}
function makeSound(animal: Animal) {
  if (animal instanceof Dog) {
    animal.bark();
  } else if (animal instanceof Cat) {
    animal.meow();
  }
}
```

---

**自定义类型守卫：使用类型谓词**
`function <函数>(<参数>): <参数> is <类型> { ... }`

```typescript
interface Person {
  name: string;
  age: number;
}
interface Animal {
  species: string;
  sound: string;
}
type LivingBeing = Person | Animal;
function isPerson(being: LivingBeing): being is Person {
  return 'name' in being && 'age' in being;
}
function processBeing(being: LivingBeing) {
  if (isPerson(being)) {
    console.log(`Person: ${being.name}, ${being.age} years old`);
  } else {
    console.log(`Animal: ${being.species}, makes ${being.sound}`);
  }
}
```

**判别式联合类型：使用共同属性区分**
`type <类型> = { kind: '<值1>'; ... } | { kind: '<值2>'; ... }`

```typescript
interface Square {
  kind: 'square';
  size: number;
}
interface Rectangle {
  kind: 'rectangle';
  width: number;
  height: number;
}
interface Circle {
  kind: 'circle';
  radius: number;
}
type Shape = Square | Rectangle | Circle;
function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'square':
      return shape.size ** 2;
    case 'rectangle':
      return shape.width * shape.height;
    case 'circle':
      return Math.PI * shape.radius ** 2;
    default:
      const exhaustiveCheck: never = shape;
      return 0;
  }
}
```

## 类型推断

**类型推断：TypeScript 自动推断类型**
`let <变量> = <值>`

```typescript
// 变量类型推断
let num = 42;        // 推断为 number
let str = 'Hello';   // 推断为 string
let isDone = true;   // 推断为 boolean
let arr = ['a', 'b', 'c'];  // 推断为 string[]
let obj = { name: 'John', age: 30 };  // 推断为 { name: string; age: number }

// 函数返回类型推断
function add(a: number, b: number) {
  return a + b;  // 返回类型推断为 number
}

// 泛型类型推断
function identity<T>(value: T): T {
  return value;
}
let num2 = identity(42);        // T 推断为 number
let str2 = identity('Hello');   // T 推断为 string

// 上下文类型推断
const names = ['John', 'Jane', 'Bob'];
names.forEach((name) => {
  console.log(name.toUpperCase());  // name 类型推断为 string
});
```
