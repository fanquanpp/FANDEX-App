# 函数与泛型

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 函数声明

**函数声明：使用 function 关键字**
`function <函数名>(<参数>: <类型>): <返回类型> { ... }`

```typescript
// 基本函数声明
function add(a: number, b: number): number {
  return a + b;
}
// 函数表达式
const multiply = function (a: number, b: number): number {
  return a * b;
};
// 箭头函数
const subtract = (a: number, b: number): number => a - b;
// 调用函数
console.log(add(5, 3)); // 8
console.log(multiply(5, 3)); // 15
console.log(subtract(5, 3)); // 2
```

## 函数类型

**函数类型：定义函数的签名**
`type <类型> = (<参数>: <类型>) => <返回类型>`
`interface <接口名> { (<参数>: <类型>): <返回类型> }`

```typescript
// 函数类型别名
type MathOperation = (a: number, b: number) => number;
// 接口形式
interface MathFunction {
  (a: number, b: number): number;
}
// 使用函数类型
const add: MathOperation = (a, b) => a + b;
const multiply: MathFunction = (a, b) => a * b;
// 作为参数类型
function calculate(op: MathOperation, a: number, b: number): number {
  return op(a, b);
}
console.log(calculate(add, 10, 5)); // 15
console.log(calculate(multiply, 10, 5)); // 50
```

## 可选参数与默认参数

**可选参数：使用 `?` 标记**
`function <函数>(<参数>?: <类型>): ...`

```typescript
// 可选参数
function greet(name: string, greeting?: string): string {
  return `${greeting || 'Hello'}, ${name}!`;
}
console.log(greet('Alice')); // Hello, Alice!
console.log(greet('Bob', 'Hi')); // Hi, Bob!
```

---

**默认参数：为参数提供默认值**
`function <函数>(<参数>: <类型> = <默认值>): ...`

```typescript
// 默认参数
function greet(name: string, greeting: string = 'Hello'): string {
  return `${greeting}, ${name}!`;
}
console.log(greet('Alice')); // Hello, Alice!
console.log(greet('Bob', 'Hi')); // Hi, Bob!
// 默认参数与可选参数结合
function createUser(name: string, age: number = 18, role: string = 'user') {
  return { name, age, role };
}
console.log(createUser('Alice')); // { name: 'Alice', age: 18, role: 'user' }
console.log(createUser('Bob', 25, 'admin')); // { name: 'Bob', age: 25, role: 'admin' }
```

## 剩余参数

**剩余参数：使用 `...` 收集多个参数**
`function <函数>(...<参数>: <类型>[]): ...`

```typescript
// 剩余参数
function sum(...numbers: number[]): number {
  return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3)); // 6
console.log(sum(1, 2, 3, 4, 5)); // 15
// 剩余参数与其他参数结合
function greetAll(greeting: string, ...names: string[]): string {
  return names.map(name => `${greeting}, ${name}!`).join(' ');
}
console.log(greetAll('Hello', 'Alice', 'Bob', 'Charlie'));
// Hello, Alice! Hello, Bob! Hello, Charlie!
```

## 函数重载

**函数重载：为同一函数提供多个类型签名**
`function <函数>(<参数>: <类型1>): <返回类型1>`
`function <函数>(<参数>: <类型2>): <返回类型2>`
`function <函数>(<参数>: <类型1> | <类型2>): <返回类型1> | <返回类型2> { ... }`

```typescript
// 函数重载签名
function process(input: string): string;
function process(input: number): number;
function process(input: boolean): boolean;
// 实现
function process(input: string | number | boolean): string | number | boolean {
  if (typeof input === 'string') {
    return input.toUpperCase();
  } else if (typeof input === 'number') {
    return input * 2;
  } else {
    return !input;
  }
}
// 使用
console.log(process('hello')); // HELLO
console.log(process(42)); // 84
console.log(process(true)); // false
```

## 泛型函数

**泛型函数：使用类型参数**
`function <函数><T>(<参数>: T): T { ... }`

```typescript
// 基本泛型函数
function identity<T>(value: T): T {
  return value;
}
// 使用泛型函数
const num = identity<number>(42); // 42
const str = identity<string>('Hello'); // Hello
// 类型推断
const inferred = identity(true); // 推断 T 为 boolean
// 多类型参数
function pair<T, U>(first: T, second: U): [T, U] {
  return [first, second];
}
const p = pair<string, number>('Age', 30); // ['Age', 30]
// 泛型与数组
function firstElement<T>(arr: T[]): T {
  return arr[0];
}
console.log(firstElement([1, 2, 3])); // 1
console.log(firstElement(['a', 'b', 'c'])); // a
```

## 泛型接口

**泛型接口：接口中使用类型参数**
`interface <接口名><T> { ... }`

```typescript
// 泛型接口
interface Box<T> {
  value: T;
}
// 使用泛型接口
const numberBox: Box<number> = { value: 42 };
const stringBox: Box<string> = { value: 'Hello' };
// 泛型函数接口
interface Transformer<T, U> {
  (input: T): U;
}
// 实现
const stringToNumber: Transformer<string, number> = (str) => str.length;
console.log(stringToNumber('hello')); // 5
```

## 泛型类

**泛型类：类中使用类型参数**
`class <类名><T> { ... }`

```typescript
// 泛型类
class Stack<T> {
  private items: T[] = [];
  push(item: T): void {
    this.items.push(item);
  }
  pop(): T | undefined {
    return this.items.pop();
  }
  peek(): T | undefined {
    return this.items[this.items.length - 1];
  }
  size(): number {
    return this.items.length;
  }
}
// 使用泛型类
const numberStack = new Stack<number>();
numberStack.push(1);
numberStack.push(2);
console.log(numberStack.pop()); // 2
const stringStack = new Stack<string>();
stringStack.push('Hello');
stringStack.push('World');
console.log(stringStack.pop()); // World
```

## 泛型约束

**泛型约束：限制类型参数的范围**
`function <函数><T extends <约束>>(<参数>: T): T { ... }`

```typescript
// 泛型约束：要求类型具有 length 属性
interface Lengthwise {
  length: number;
}
function logLength<T extends Lengthwise>(arg: T): void {
  console.log(arg.length);
}
logLength('Hello'); // 5
logLength([1, 2, 3]); // 3
// logLength(42); // 错误：number 没有 length 属性
// 使用 keyof 约束
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}
const person = { name: 'Alice', age: 30, city: 'New York' };
console.log(getProperty(person, 'name')); // Alice
console.log(getProperty(person, 'age')); // 30
// getProperty(person, 'email'); // 错误：'email' 不是 person 的属性
```

## 泛型默认类型

**泛型默认类型：为类型参数提供默认值**
`function <函数><T = <默认类型>>(<参数>: T): T { ... }`

```typescript
// 泛型默认类型
function createArray<T = string>(length: number, value: T): T[] {
  return Array(length).fill(value);
}
// 使用默认类型
const strings = createArray(3, 'Hello'); // string[]
const numbers = createArray<number>(3, 42); // number[]
// 泛型接口默认类型
interface ApiResponse<T = any> {
  data: T;
  status: number;
  message: string;
}
// 使用默认类型
const response1: ApiResponse = {
  data: 'Hello',
  status: 200,
  message: 'Success',
};
const response2: ApiResponse<number[]> = {
  data: [1, 2, 3],
  status: 200,
  message: 'Success',
};
```

## 条件类型

**条件类型：根据类型关系选择不同类型**
`type <类型> = T extends U ? X : Y`

```typescript
// 条件类型
type IsString<T> = T extends string ? true : false;
type A = IsString<string>; // true
type B = IsString<number>; // false
// 使用条件类型的函数
function isString<T>(value: T): T extends string ? true : false {
  return (typeof value === 'string') as any;
}
console.log(isString('hello')); // true
console.log(isString(42)); // false
```

## 泛型工具类型

**常用泛型工具类型**
`type <类型> = Partial<T>`
`type <类型> = Readonly<T>`
`type <类型> = Pick<T, K>`
`type <类型> = Omit<T, K>`

```typescript
interface Todo {
  title: string;
  description: string;
  completed: boolean;
}
// Partial：所有属性变为可选
type PartialTodo = Partial<Todo>;
const partial: PartialTodo = { title: 'Learn TypeScript' };
// Readonly：所有属性变为只读
type ReadonlyTodo = Readonly<Todo>;
const readonly: ReadonlyTodo = {
  title: 'Learn TypeScript',
  description: 'Learn TypeScript',
  completed: false,
};
// readonly.title = 'New Title'; // 错误：只读属性
// Pick：选择部分属性
type TodoPreview = Pick<Todo, 'title' | 'completed'>;
const preview: TodoPreview = {
  title: 'Learn TypeScript',
  completed: false,
};
// Omit：排除部分属性
type TodoInfo = Omit<Todo, 'completed'>;
const info: TodoInfo = {
  title: 'Learn TypeScript',
  description: 'Learn TypeScript',
};
```
