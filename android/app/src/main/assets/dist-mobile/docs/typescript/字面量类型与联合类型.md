# 字面量类型与联合类型

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 字符串字面量类型

**字符串字面量类型：限制值为特定字符串**
`type <类型> = '<值1>' | '<值2>' | ...`

```typescript
// 基本字符串字面量类型
type Direction = 'North' | 'South' | 'East' | 'West';
let move: Direction = 'North';
// move = 'Up'; // 错误：'Up' 不是 Direction 类型
// 应用：配置选项
type LogLevel = 'debug' | 'info' | 'warn' | 'error';
function log(level: LogLevel, message: string): void {
  console.log(`[${level.toUpperCase()}] ${message}`);
}
log('info', 'Application started');
log('error', 'Something went wrong');
// log('trace', 'Detailed trace'); // 错误：'trace' 不是 LogLevel 类型
```

## 数字字面量类型

**数字字面量类型：限制值为特定数字**
`type <类型> = <数字1> | <数字2> | ...`

```typescript
// 基本数字字面量类型
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;
let roll: DiceRoll = 4;
// roll = 7; // 错误：7 不是 DiceRoll 类型
// 应用：状态码
type HttpStatusCode = 200 | 404 | 500;
function handleResponse(code: HttpStatusCode): string {
  switch (code) {
    case 200:
      return 'OK';
    case 404:
      return 'Not Found';
    case 500:
      return 'Internal Server Error';
  }
}
console.log(handleResponse(200)); // OK
console.log(handleResponse(404)); // Not Found
```

## 布尔字面量类型

**布尔字面量类型：限制值为 true 或 false**
`type <类型> = true | false`

```typescript
// 布尔字面量类型
type OnlyTrue = true;
type OnlyFalse = false;
let isTrue: OnlyTrue = true;
let isFalse: OnlyFalse = false;
// isTrue = false; // 错误：OnlyTrue 只能是 true
// 应用：配置选项
interface Config {
  readonly: true;
  cache: boolean;
}
const config: Config = {
  readonly: true,
  cache: false,
};
```

## 联合类型

**基本联合类型：使用 `|` 表示值可以是其中之一**
`type <类型> = <类型1> | <类型2>`

```typescript
// 基本联合类型
type ID = string | number;
let userId: ID = '123';
userId = 456;
// 联合类型与类型守卫
function processId(id: ID) {
  if (typeof id === 'string') {
    console.log(`String ID: ${id.toUpperCase()}`);
  } else {
    console.log(`Number ID: ${id.toFixed(2)}`);
  }
}
processId('abc'); // String ID: ABC
processId(42); // Number ID: 42.00
```

---

**字面量联合类型：组合多个字面量**
`type <类型> = '<值1>' | '<值2>' | <数字> | ...`

```typescript
// 混合字面量联合类型
type ResponseCode = 200 | 404 | 500 | 'unknown';
function handleCode(code: ResponseCode): string {
  if (code === 200) return 'OK';
  if (code === 404) return 'Not Found';
  if (code === 500) return 'Internal Server Error';
  return 'Unknown';
}
console.log(handleCode(200)); // OK
console.log(handleCode('unknown')); // Unknown
// 应用：UI 状态
type ButtonState = 'default' | 'hover' | 'active' | 'disabled';
function getButtonColor(state: ButtonState): string {
  const colors = {
    default: '#007bff',
    hover: '#0056b3',
    active: '#004085',
    disabled: '#6c757d',
  };
  return colors[state];
}
console.log(getButtonColor('hover')); // #0056b3
```

## 字面量推断

**字面量推断：const 断言与字面量类型**
`const <变量> = <值>`
`let <变量>: <字面量类型> = <值>`

```typescript
// let 声明：推断为宽泛类型
let str = 'hello'; // 类型: string
let num = 42; // 类型: number
let bool = true; // 类型: boolean
// const 声明：推断为字面量类型
const str2 = 'hello'; // 类型: "hello"
const num2 = 42; // 类型: 42
const bool2 = true; // 类型: true
// 对象字面量推断
const config = {
  apiUrl: 'https://api.example.com',
  timeout: 5000,
  retries: 3,
};
// config 类型: { apiUrl: string; timeout: number; retries: number }
// 使用 as const 使对象属性变为字面量类型
const config2 = {
  apiUrl: 'https://api.example.com',
  timeout: 5000,
  retries: 3,
} as const;
// config2 类型: { readonly apiUrl: "https://api.example.com"; readonly timeout: 5000; readonly retries: 3 }
```

## 判别式联合

**判别式联合：使用共同属性区分类型**
`type <类型> = { kind: '<值1>'; ... } | { kind: '<值2>'; ... }`

```typescript
// 判别式联合类型
interface Circle {
  kind: 'circle';
  radius: number;
}
interface Square {
  kind: 'square';
  sideLength: number;
}
interface Rectangle {
  kind: 'rectangle';
  width: number;
  height: number;
}
type Shape = Circle | Square | Rectangle;
// 使用判别式联合
function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    case 'rectangle':
      return shape.width * shape.height;
    default:
      const exhaustiveCheck: never = shape;
      return 0;
  }
}
console.log(getArea({ kind: 'circle', radius: 5 })); // 78.53981633974483
console.log(getArea({ kind: 'square', sideLength: 4 })); // 16
console.log(getArea({ kind: 'rectangle', width: 3, height: 6 })); // 18
```

## 类型守卫与联合类型

**typeof 类型守卫：缩小联合类型**
`if (typeof <变量> === '<类型>') { ... }`

```typescript
// typeof 类型守卫
function process(value: string | number | boolean) {
  if (typeof value === 'string') {
    return value.toUpperCase();
  } else if (typeof value === 'number') {
    return value.toFixed(2);
  } else {
    return value ? 'true' : 'false';
  }
}
console.log(process('hello')); // HELLO
console.log(process(3.14159)); // 3.14
console.log(process(true)); // true
```

---

**in 操作符类型守卫：检查属性是否存在**
`if ('<属性>' in <变量>) { ... }`

```typescript
interface Fish {
  swim(): void;
}
interface Bird {
  fly(): void;
}
type Pet = Fish | Bird;
function move(pet: Pet) {
  if ('swim' in pet) {
    pet.swim();
  } else {
    pet.fly();
  }
}
```

---

**instanceof 类型守卫：检查类实例**
`if (<变量> instanceof <类>) { ... }`

```typescript
class ErrorWithCode extends Error {
  code: number;
  constructor(message: string, code: number) {
    super(message);
    this.code = code;
  }
}
function handleError(error: Error | ErrorWithCode) {
  if (error instanceof ErrorWithCode) {
    console.log(`Error ${error.code}: ${error.message}`);
  } else {
    console.log(`Error: ${error.message}`);
  }
}
```

## 联合类型的应用

**函数参数联合类型**
`function <函数>(<参数>: <类型1> | <类型2>): ...`

```typescript
// 函数参数联合类型
function formatValue(value: string | number): string {
  if (typeof value === 'string') {
    return value.trim().toUpperCase();
  } else {
    return value.toFixed(2).padStart(10, '0');
  }
}
console.log(formatValue('  hello  ')); // HELLO
console.log(formatValue(42)); // 0000042.00
```

---

**数组元素联合类型**
`type <类型> = Array<<类型1> | <类型2>>`

```typescript
// 数组元素联合类型
type MixedArray = Array<string | number>;
const mixed: MixedArray = ['hello', 42, 'world', 100];
// 过滤特定类型
function filterStrings(arr: MixedArray): string[] {
  return arr.filter((item): item is string => typeof item === 'string');
}
console.log(filterStrings(mixed)); // ['hello', 'world']
```

## 可辨识联合模式

**状态机模式：使用判别式联合建模状态**
`type <状态> = { status: '<值1>'; ... } | { status: '<值2>'; ... }`

```typescript
// 状态机模式
type RequestState =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: any }
  | { status: 'error'; error: string };
// 处理不同状态
function handleRequest(state: RequestState): string {
  switch (state.status) {
    case 'idle':
      return 'Ready to start';
    case 'loading':
      return 'Loading...';
    case 'success':
      return `Data: ${JSON.stringify(state.data)}`;
    case 'error':
      return `Error: ${state.error}`;
  }
}
console.log(handleRequest({ status: 'idle' })); // Ready to start
console.log(handleRequest({ status: 'loading' })); // Loading...
console.log(handleRequest({ status: 'success', data: { id: 1 } })); // Data: {"id":1}
console.log(handleRequest({ status: 'error', error: 'Network error' })); // Error: Network error
```

## 模板字面量类型

**模板字面量类型：基于字符串模板生成类型**
`type <类型> = \`<前缀>\${<类型>}\``

```typescript
// 基本模板字面量类型
type HelloWorld = `hello ${string}`;
let greeting: HelloWorld = 'hello world';
let greeting2: HelloWorld = 'hello typescript';
// greeting3: HelloWorld = 'hi world'; // 错误
// 组合字面量类型
type Direction = 'left' | 'right';
type Margin = `margin-${Direction}`;
let margin: Margin = 'margin-left';
let margin2: Margin = 'margin-right';
```
