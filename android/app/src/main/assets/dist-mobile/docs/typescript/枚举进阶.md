# 枚举进阶

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 数字枚举

**数字枚举：默认从 0 开始递增**
`enum <枚举名> { <成员1>, <成员2>, ... }`

```typescript
// 基本数字枚举
enum Direction {
  Up,    // 0
  Down,  // 1
  Left,  // 2
  Right, // 3
}
let dir: Direction = Direction.Up;
console.log(dir); // 0
// 自定义起始值
enum StatusCode {
  OK = 200,
  NotFound = 404,
  InternalError = 500,
}
console.log(StatusCode.OK); // 200
// 自定义递增值
enum Permission {
  Read = 1,
  Write = 2,
  Execute = 4,
}
console.log(Permission.Read); // 1
console.log(Permission.Write); // 2
console.log(Permission.Execute); // 4
```

---

**反向映射：数字枚举支持从值获取键**
`<枚举名>[<值>]`

```typescript
enum Direction {
  Up,
  Down,
  Left,
  Right,
}
// 正向映射
console.log(Direction.Up); // 0
// 反向映射
console.log(Direction[0]); // "Up"
console.log(Direction[1]); // "Down"
```

## 字符串枚举

**字符串枚举：成员值为字符串**
`enum <枚举名> { <成员1> = '<值1>', <成员2> = '<值2>' }`

```typescript
// 字符串枚举
enum Status {
  Active = 'ACTIVE',
  Inactive = 'INACTIVE',
  Pending = 'PENDING',
}
let status: Status = Status.Active;
console.log(status); // "ACTIVE"
// 字符串枚举没有反向映射
console.log(Status['ACTIVE']); // 错误
console.log(Status.Active); // "ACTIVE"
```

## 异构枚举

**异构枚举：混合数字和字符串值**
`enum <枚举名> { <成员1> = <数字>, <成员2> = '<字符串>' }`

```typescript
// 异构枚举（不推荐使用）
enum BooleanLikeHeterogeneousEnum {
  No = 0,
  Yes = 'YES',
}
console.log(BooleanLikeHeterogeneousEnum.No); // 0
console.log(BooleanLikeHeterogeneousEnum.Yes); // "YES"
```

## 计算成员

**计算成员：使用表达式作为枚举值**
`enum <枚举名> { <成员> = <表达式> }`

```typescript
// 计算成员
enum FileAccess {
  None = 0,
  Read = 1 << 1,    // 2
  Write = 1 << 2,   // 4
  Execute = 1 << 3, // 8
  All = Read | Write | Execute, // 14
}
console.log(FileAccess.Read); // 2
console.log(FileAccess.Write); // 4
console.log(FileAccess.All); // 14
// 使用枚举进行位运算
const permission = FileAccess.Read | FileAccess.Write;
console.log(permission); // 6
console.log(permission & FileAccess.Read); // 2（有读权限）
console.log(permission & FileAccess.Execute); // 0（无执行权限）
```

## 常量枚举

**const enum：编译时内联，不生成代码**
`const enum <枚举名> { <成员1>, <成员2>, ... }`

```typescript
// 常量枚举
const enum Direction {
  Up,
  Down,
  Left,
  Right,
}
// 使用常量枚举
let directions: Direction[] = [Direction.Up, Direction.Down, Direction.Left, Direction.Right];
// 编译后：let directions = [0, 1, 2, 3];
console.log(directions); // [0, 1, 2, 3]
```

## 枚举成员类型

**枚举成员作为类型**
`type <类型> = <枚举名>.<成员>`

```typescript
enum ShapeKind {
  Circle,
  Square,
  Triangle,
}
interface Shape {
  kind: ShapeKind.Circle; // 字面量类型
  radius: number;
}
const circle: Shape = {
  kind: ShapeKind.Circle,
  radius: 5,
};

//   kind: ShapeKind.Square, // 错误：类型不匹配
//   radius: 5,
// };
```

## 运行时枚举

**运行时枚举：枚举是实际对象**
`enum <枚举名> { ... }`

```typescript
enum LogLevel {
  ERROR,
  WARN,
  INFO,
  DEBUG,
}
// 运行时使用
function log(level: LogLevel, message: string): void {
  console.log(`[${LogLevel[level]}] ${message}`);
}
log(LogLevel.ERROR, 'Something went wrong'); // [ERROR] Something went wrong
log(LogLevel.INFO, 'Information message'); // [INFO] Information message
// 遍历枚举
for (const level in LogLevel) {
  if (isNaN(Number(level))) {
    console.log(level); // ERROR, WARN, INFO, DEBUG
  }
}
```

## 枚举与联合类型

**枚举与联合类型组合**
`type <类型> = <枚举名>.<成员1> | <枚举名>.<成员2>`

```typescript
enum Color {
  Red,
  Green,
  Blue,
}
// 使用枚举成员作为联合类型
type PrimaryColor = Color.Red | Color.Green | Color.Blue;
function getColorName(color: PrimaryColor): string {
  switch (color) {
    case Color.Red:
      return 'Red';
    case Color.Green:
      return 'Green';
    case Color.Blue:
      return 'Blue';
    default:
      return 'Unknown';
  }
}
console.log(getColorName(Color.Red)); // Red
```

## 枚举方法

**为枚举添加方法：使用命名空间合并**
`namespace <枚举名> { export function <方法>(): ... }`

```typescript
enum Weekday {
  Monday,
  Tuesday,
  Wednesday,
  Thursday,
  Friday,
  Saturday,
  Sunday,
}
// 使用命名空间为枚举添加方法
namespace Weekday {
  export function isWeekend(day: Weekday): boolean {
    return day === Weekday.Saturday || day === Weekday.Sunday;
  }
  export function isBusinessDay(day: Weekday): boolean {
    return !isWeekend(day);
  }
}
console.log(Weekday.isWeekend(Weekday.Saturday)); // true
console.log(Weekday.isWeekend(Weekday.Monday)); // false
console.log(Weekday.isBusinessDay(Weekday.Friday)); // true
```

## 枚举与映射

**枚举与映射：使用 Record 创建枚举映射**
`type <类型> = Record<<枚举>, <值类型>>`

```typescript
enum HttpStatus {
  OK = 200,
  NotFound = 404,
  InternalServerError = 500,
}
// 使用 Record 创建枚举映射
const statusMessages: Record<HttpStatus, string> = {
  [HttpStatus.OK]: 'OK',
  [HttpStatus.NotFound]: 'Not Found',
  [HttpStatus.InternalServerError]: 'Internal Server Error',
};
console.log(statusMessages[HttpStatus.OK]); // OK
console.log(statusMessages[HttpStatus.NotFound]); // Not Found
```

## 枚举工具类型

**枚举工具类型：提取键和值**
`type <类型> = keyof typeof <枚举>`
`type <类型> = typeof <枚举>[keyof typeof <枚举>]`

```typescript
enum Color {
  Red = 'RED',
  Green = 'GREEN',
  Blue = 'BLUE',
}
// 提取枚举键
type ColorKey = keyof typeof Color; // 'Red' | 'Green' | 'Blue'
// 提取枚举值
type ColorValue = typeof Color[keyof typeof Color]; // 'RED' | 'GREEN' | 'BLUE'
// 使用
const colorName: ColorKey = 'Red';
const colorValue: ColorValue = 'RED';
console.log(colorName, colorValue); // Red RED
```

## 枚举与对象字面量

**对象字面量替代枚举**
`const <对象> = { <键>: <值> } as const`

```typescript
// 使用对象字面量替代枚举
const Status = {
  Active: 'ACTIVE',
  Inactive: 'INACTIVE',
  Pending: 'PENDING',
} as const;
// 提取类型
type Status = typeof Status[keyof typeof Status]; // 'ACTIVE' | 'INACTIVE' | 'PENDING'
type StatusKey = keyof typeof Status; // 'Active' | 'Inactive' | 'Pending'
// 使用
const status: Status = Status.Active;
console.log(status); // ACTIVE
```
