# 泛型约束与默认值

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 泛型约束基础

**extends 约束：限制类型参数**
`function <函数><T extends <约束>>(<参数>: T): T`

```typescript
// 基本泛型约束
interface Lengthwise {
  length: number;
}
function logLength<T extends Lengthwise>(arg: T): void {
  console.log(arg.length);
}
logLength('Hello'); // 5
logLength([1, 2, 3]); // 3
// logLength(42); // 错误：number 没有 length 属性
```

---

## keyof 约束

**keyof 约束：限制键为对象属性**
`function <函数><T, K extends keyof T>(<参数>: T, <键>: K): T[K]`

```typescript
// 使用 keyof 约束
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}
const person = { name: 'Alice', age: 30, city: 'New York' };
console.log(getProperty(person, 'name')); // Alice
console.log(getProperty(person, 'age')); // 30
// getProperty(person, 'email'); // 错误：'email' 不是 person 的属性
// 设置属性
function setProperty<T, K extends keyof T>(obj: T, key: K, value: T[K]): void {
  obj[key] = value;
}
setProperty(person, 'name', 'Bob');
console.log(person.name); // Bob
```

## 类类型约束

**类类型约束：使用构造函数签名**
`function <函数><T extends <类>>(c: new () => T): T`

```typescript
// 类类型约束
class BeeKeeper {
  hasMask: boolean = true;
}
class ZooKeeper {
  nametag: string = 'ZooKeeper';
}
class Animal {
  numLegs: number = 4;
}
class Bee extends Animal {
  keeper: BeeKeeper = new BeeKeeper();
}
class Lion extends Animal {
  keeper: ZooKeeper = new ZooKeeper();
}
// 创建实例的泛型函数
function createInstance<A extends Animal>(c: new () => A): A {
  return new c();
}
const bee = createInstance(Bee);
const lion = createInstance(Lion);
console.log(bee.keeper.hasMask); // true
console.log(lion.keeper.nametag); // ZooKeeper
```

## 多重约束

**多重约束：使用交叉类型**
`function <函数><T extends <约束1> & <约束2>>(<参数>: T): T`

```typescript
// 多重约束
interface Identifiable {
  id: number;
}
interface Nameable {
  name: string;
}
function process<T extends Identifiable & Nameable>(item: T): void {
  console.log(`ID: ${item.id}, Name: ${item.name}`);
}
// 使用
const user = { id: 1, name: 'Alice', age: 30 };
process(user); // ID: 1, Name: Alice
```

## 泛型默认类型

**泛型默认类型：为类型参数提供默认值**
`function <函数><T = <默认类型>>(<参数>: T): T`

```typescript
// 泛型默认类型
function createArray<T = string>(length: number, value: T): T[] {
  return Array(length).fill(value);
}
// 使用默认类型
const strings = createArray(3, 'Hello'); // string[]
const numbers = createArray<number>(3, 42); // number[]
// 多个默认类型
function pair<T = string, U = number>(first: T, second: U): [T, U] {
  return [first, second];
}
const p1 = pair('Age', 30); // [string, number]
const p2 = pair<boolean, string>(true, 'Yes'); // [boolean, string]
```

## 泛型接口默认类型

**泛型接口默认类型**
`interface <接口名><T = <默认类型>> { ... }`

```typescript
// 泛型接口默认类型
interface Box<T = string> {
  value: T;
}
// 使用默认类型
const stringBox: Box = { value: 'Hello' };
const numberBox: Box<number> = { value: 42 };
// 复杂默认类型
interface ApiResponse<T = any, E = Error> {
  data: T;
  error: E | null;
}
const response1: ApiResponse = {
  data: 'Hello',
  error: null,
};
const response2: ApiResponse<number[]> = {
  data: [1, 2, 3],
  error: null,
};
```

## 泛型类默认类型

**泛型类默认类型**
`class <类名><T = <默认类型>> { ... }`

```typescript
// 泛型类默认类型
class Container<T = string> {
  private items: T[] = [];
  add(item: T): void {
    this.items.push(item);
  }
  get(index: number): T | undefined {
    return this.items[index];
  }
}
// 使用默认类型
const stringContainer = new Container();
stringContainer.add('Hello');
console.log(stringContainer.get(0)); // Hello
// 指定类型
const numberContainer = new Container<number>();
numberContainer.add(42);
console.log(numberContainer.get(0)); // 42
```

## 条件类型约束

**条件类型约束：根据条件选择约束**
`type <类型><T> = T extends <约束> ? <类型1> : <类型2>`

```typescript
// 条件类型约束
type NonNullable<T> = T extends null | undefined ? never : T;
type A = NonNullable<string | null>; // string
type B = NonNullable<number | undefined>; // number
// 排除特定类型
type Exclude<T, U> = T extends U ? never : T;
type C = Exclude<'a' | 'b' | 'c', 'a'>; // 'b' | 'c'
// 提取特定类型
type Extract<T, U> = T extends U ? T : never;
type D = Extract<string | number | boolean, string | number>; // string | number
```

## infer 关键字

**infer：在条件类型中推断类型**
`type <类型><T> = T extends Array<infer U> ? U : T`

```typescript
// 提取数组元素类型
type UnpackArray<T> = T extends Array<infer U> ? U : T;
type A = UnpackArray<string[]>; // string
type B = UnpackArray<number[]>; // number
type C = UnpackArray<string>; // string
// 提取函数返回类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : any;
type R = ReturnType<() => string>; // string
// 提取函数参数类型
type Parameters<T> = T extends (...args: infer P) => any ? P : never;
type P = Parameters<(a: string, b: number) => void>; // [string, number]
```

## 递归条件类型

**递归条件类型：处理嵌套结构**
`type <类型><T> = T extends Array<infer U> ? <递归调用> : T`

```typescript
// 递归展平数组类型
type DeepFlatten<T> = T extends Array<infer U> ? DeepFlatten<U> : T;
type A = DeepFlatten<number[][][]>; // number
// 递归提取 Promise 类型
type UnpackPromise<T> = T extends Promise<infer U> ? UnpackPromise<U> : T;
type B = UnpackPromise<Promise<Promise<string>>>; // string
```

## 泛型工具类型

**内置泛型工具类型**
`type <类型> = Partial<T>`
`type <类型> = Required<T>`
`type <类型> = Readonly<T>`
`type <类型> = Record<K, T>`

```typescript
interface Todo {
  title: string;
  description: string;
  completed?: boolean;
}
// Partial：所有属性变为可选
type PartialTodo = Partial<Todo>;
// Required：所有属性变为必需
type RequiredTodo = Required<Todo>;
// Readonly：所有属性变为只读
type ReadonlyTodo = Readonly<Todo>;
// Record：键值对类型
type TodoMap = Record<string, Todo>;
// Pick：选择部分属性
type TodoPreview = Pick<Todo, 'title' | 'completed'>;
// Omit：排除部分属性
type TodoInfo = Omit<Todo, 'completed'>;
```

## 自定义泛型工具类型

**自定义工具类型：组合泛型约束**
`type <类型><T> = ...`

```typescript
// 自定义工具类型
// 获取对象类型的键
type GetKeys<T> = keyof T;
// 获取对象类型的值类型
type GetValues<T> = T[keyof T];
// 将所有属性变为可空
type Nullable<T> = { [P in keyof T]: T[P] | null };
// 反转属性的可选性
type Mutable<T> = { -readonly [P in keyof T]: T[P] };
// 深度 Partial
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};
// 深度 Readonly
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object ? DeepReadonly<T[P]> : T[P];
};
```
