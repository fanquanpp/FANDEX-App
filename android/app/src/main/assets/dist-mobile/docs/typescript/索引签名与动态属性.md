# 索引签名与动态属性

> **符号约定**：`< >` 必填参数 | `[ ]` 可选参数

---

## 索引签名基础

**字符串索引签名：定义任意字符串键**
`interface <接口名> { [key: string]: <值类型> }`

```typescript
// 字符串索引签名
interface StringMap {
  [key: string]: string;
}
const colors: StringMap = {
  red: '#FF0000',
  green: '#00FF00',
  blue: '#0000FF',
};
// 可以添加任意字符串键
colors.yellow = '#FFFF00';
console.log(colors.red); // #FF0000
console.log(colors.yellow); // #FFFF00
```

---

**数字索引签名：定义数字键**
`interface <接口名> { [index: number]: <值类型> }`

```typescript
// 数字索引签名
interface NumberArray {
  [index: number]: string;
}
const arr: NumberArray = ['a', 'b', 'c'];
console.log(arr[0]); // a
console.log(arr[1]); // b
// 可以添加任意数字键
arr[10] = 'k';
console.log(arr[10]); // k
```

## 混合索引签名

**字符串和数字索引签名组合**
`interface <接口名> { [key: string]: <类型1>; [index: number]: <类型2> }`

```typescript
// 混合索引签名
interface MixedMap {
  [key: string]: string | number;
  [index: number]: string; // 数字索引返回类型必须是字符串索引返回类型的子类型
  length: number; // 具体属性类型必须与索引签名兼容
}
const mixed: MixedMap = {
  name: 'Alice',
  age: 30,
  length: 2,
};
console.log(mixed.name); // Alice
console.log(mixed.age); // 30
console.log(mixed.length); // 2
```

## 只读索引签名

**readonly 索引签名：防止修改**
`interface <接口名> { readonly [key: string]: <值类型> }`

```typescript
// 只读索引签名
interface ReadonlyMap {
  readonly [key: string]: number;
}
const config: ReadonlyMap = {
  timeout: 5000,
  retries: 3,
};
console.log(config.timeout); // 5000
// config.timeout = 10000; // 错误：只读属性
// config.newKey = 100; // 错误：只读属性
```

## 索引签名与具体属性

**索引签名与具体属性共存**
`interface <接口名> { [key: string]: <值类型>; <具体属性>: <类型> }`

```typescript
// 索引签名与具体属性
interface User {
  [key: string]: string | number | boolean | undefined;
  name: string;
  age: number;
  isActive: boolean;
}
const user: User = {
  name: 'Alice',
  age: 30,
  isActive: true,
  email: 'alice@example.com', // 额外属性
  phone: '123-456-7890',      // 额外属性
};
console.log(user.name);  // Alice
console.log(user.email); // alice@example.com
```

## Record 工具类型

**Record：键值对类型**
`type <类型> = Record<<键类型>, <值类型>>`

```typescript
// Record 工具类型
type StringNumberMap = Record<string, number>;
const map: StringNumberMap = {
  one: 1,
  two: 2,
  three: 3,
};
// 使用字面量类型作为键
type Page = 'home' | 'about' | 'contact';
type PageTitle = Record<Page, string>;
const titles: PageTitle = {
  home: 'Home Page',
  about: 'About Us',
  contact: 'Contact Us',
};
// 使用枚举作为键
enum HttpStatus {
  OK = 200,
  NotFound = 404,
  InternalServerError = 500,
}
type StatusMessage = Record<HttpStatus, string>;
const messages: StatusMessage = {
  [HttpStatus.OK]: 'OK',
  [HttpStatus.NotFound]: 'Not Found',
  [HttpStatus.InternalServerError]: 'Internal Server Error',
};
```

## 动态属性访问

**动态属性访问：使用变量作为键**
`<对象>[<键变量>]`

```typescript
interface User {
  name: string;
  age: number;
  email: string;
}
const user: User = {
  name: 'Alice',
  age: 30,
  email: 'alice@example.com',
};
// 动态属性访问
const key: keyof User = 'name';
console.log(user[key]); // Alice
// 使用函数动态访问属性
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}
console.log(getProperty(user, 'name')); // Alice
console.log(getProperty(user, 'age')); // 30
```

## 映射类型与动态属性

**映射类型：基于联合类型生成属性**
`type <类型> = { [K in <联合类型>]: <值类型> }`

```typescript
// 映射类型
type Keys = 'name' | 'age' | 'email';
type User = {
  [K in Keys]: string;
};
// 等价于:
// type User = {
//   name: string;
//   age: string;
//   email: string;
// }
const user: User = {
  name: 'Alice',
  age: '30',
  email: 'alice@example.com',
};
```

## 索引访问类型

**索引访问类型：获取属性类型**
`type <类型> = <对象类型>[<键>]`

```typescript
interface User {
  name: string;
  age: number;
  email: string;
}
// 索引访问类型
type NameType = User['name']; // string
type AgeType = User['age']; // number
// 使用 keyof 获取所有属性的类型
type UserType = User[keyof User]; // string | number
// 数组元素类型
type NumberArray = number[];
type ElementType = NumberArray[number]; // number
```

## 动态对象创建

**动态创建对象：使用计算属性名**
`const <对象> = { [<表达式>]: <值> }`

```typescript
// 计算属性名
const propName = 'dynamic_' + Date.now();
const obj = {
  [propName]: 'value',
  staticProp: 'static',
};
console.log(obj[propName]); // value
// 动态创建对象
function createObject(keys: string[], values: any[]) {
  const result: { [key: string]: any } = {};
  keys.forEach((key, index) => {
    result[key] = values[index];
  });
  return result;
}
const dynamicObj = createObject(['name', 'age', 'email'], ['Alice', 30, 'alice@example.com']);
console.log(dynamicObj.name); // Alice
console.log(dynamicObj.age); // 30
```

## 索引签名与类

**类中的索引签名**
`class <类名> { [key: string]: <值类型> }`

```typescript
// 类中的索引签名
class Config {
  [key: string]: any;
  constructor(initialConfig: Record<string, any>) {
    Object.assign(this, initialConfig);
  }
  has(key: string): boolean {
    return key in this;
  }
  get(key: string): any {
    return this[key];
  }
  set(key: string, value: any): void {
    this[key] = value;
  }
}
const config = new Config({
  apiUrl: 'https://api.example.com',
  timeout: 5000,
});
console.log(config.get('apiUrl')); // https://api.example.com
config.set('debug', true);
console.log(config.get('debug')); // true
console.log(config.has('timeout')); // true
```

## 索引签名与函数

**索引签名与函数类型**
`interface <接口名> { [key: string]: (...args: any[]) => any }`

```typescript
// 索引签名与函数
interface EventHandler {
  [event: string]: (...args: any[]) => void;
}
class EventBus {
  private handlers: EventHandler = {};
  on(event: string, handler: (...args: any[]) => void): void {
    this.handlers[event] = handler;
  }
  emit(event: string, ...args: any[]): void {
    if (this.handlers[event]) {
      this.handlers[event](...args);
    }
  }
}
const bus = new EventBus();
bus.on('click', (x: number, y: number) => {
  console.log(`Clicked at (${x}, ${y})`);
});
bus.emit('click', 100, 200); // Clicked at (100, 200)
```

## 索引签名的限制

**索引签名的限制：无法使用特定类型检查**
`interface <接口名> { [key: string]: <值类型> }`

```typescript
// 索引签名的限制
interface Dictionary {
  [key: string]: string;
}
const dict: Dictionary = {
  a: 'apple',
  b: 'banana',
  c: 'cherry',
};
// 遍历所有键
for (const key in dict) {
  console.log(`${key}: ${dict[key]}`);
}
// 检查键是否存在
console.log('a' in dict); // true
console.log('x' in dict); // false
// 获取所有键
console.log(Object.keys(dict)); // ['a', 'b', 'c']
// 获取所有值
console.log(Object.values(dict)); // ['apple', 'banana', 'cherry']
// 获取所有键值对
console.log(Object.entries(dict)); // [['a', 'apple'], ['b', 'banana'], ['c', 'cherry']]
```

## 替代方案

**使用具体接口替代索引签名**
`interface <接口名> { <属性1>: <类型>; <属性2>: <类型> }`

```typescript
// 使用具体接口替代索引签名
interface User {
  name: string;
  age: number;
  email: string;
}
// 使用 Partial 使所有属性可选
type PartialUser = Partial<User>;
const partial: PartialUser = {
  name: 'Alice',
};
// 使用 Record 创建键值对
type UserMap = Record<string, User>;
const userMap: UserMap = {
  user1: { name: 'Alice', age: 30, email: 'alice@example.com' },
  user2: { name: 'Bob', age: 25, email: 'bob@example.com' },
};
// 使用 Map 类
const userMapInstance = new Map<string, User>();
userMapInstance.set('user1', { name: 'Alice', age: 30, email: 'alice@example.com' });
userMapInstance.set('user2', { name: 'Bob', age: 25, email: 'bob@example.com' });
console.log(userMapInstance.get('user1')); // { name: 'Alice', age: 30, email: 'alice@example.com' }
```
